\begin{enumerate}
    \item Oefening:
    $$
    z^3 + 5i z^2 + z + 5i
    $$

    \item Ontbinding in factoren:
    $$
    z^2(z + 5i) + 1(z + 5i) = 0
    $$
    $$
    (z^2 + 1)(z + 5i) = 0
    $$
    \begin{itemize}
        \item \( z + 5i = 0 \Rightarrow z = -5i \)
        \item \( z^2 + 1 = 0 \Rightarrow z = \pm i \)
    \end{itemize}

    \item Nulpunten in \(\mathbb{C}\):
    $$
    z = -5i, \quad z = i, \quad z = -i
    $$

    \item Eindresultaat \(\mathbb{C}[z]\):
    $$
    (z + 5i)(z - i)(z + i)
    $$
\end{enumerate}
