$ r = \sin\theta + \cos\theta $\\

$ \operatorname{dom} = \mathbb{R} $\\

$ p = 2\pi $\\

$ \rightarrow \operatorname{BD} = [0 , 2\pi] $\\

$ r = 0 \Leftrightarrow \sin\theta + \cos\theta = 0 \Leftrightarrow \sin\theta = -\cos\theta \Leftrightarrow \theta = \dfrac{3\pi}{4} + k\pi,  \forall k \in \mathbb{Z} $\\

$ \rightarrow \left\{\dfrac{3\pi}{4}, \dfrac{7\pi}{4}\right\} $\\

$ r' = 0 \Leftrightarrow  \cos\theta - \sin\theta = 0 \Leftrightarrow \cos\theta =  \sin\theta \Leftrightarrow \theta = \dfrac{\pi}{4} + k\pi,  \forall k \in \mathbb{Z} $\\

$ \rightarrow \left\{ \dfrac{\pi}{4}, \dfrac{5\pi}{4}\right\} $\\
