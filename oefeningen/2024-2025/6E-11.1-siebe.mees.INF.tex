\begin{enumerate}
  \item $f(x) = \cosh x + \cos x - \dfrac{x^4}{12}$
  \item $f'(x) = \sinh x - \sin x - \dfrac{x^3}{3} \Rightarrow f'(0) = 0$
  \item $f''(x) = \cosh x - \cos x - x^2 \Rightarrow f''(0) = 1 - 1 - 0 = 0$
  \item $f^{(3)}(x) = \sinh x + \sin x - 2x \Rightarrow f^{(3)}(0) = 0$
  \item $f^{(4)}(x) = \cosh x + \cos x - 2 \Rightarrow f^{(4)}(0) = 1 + 1 - 2 = 0$
  \item $f^{(5)}(x) = \sinh x - \sin x \Rightarrow f^{(5)}(0) = 0$
  \item $f^{(6)}(x) = \cosh x - \cos x \Rightarrow f^{(6)}(0) = 1 - 1 = 0$
  \item $f^{(7)}(x) = \sinh x + \sin x \Rightarrow f^{(7)}(0) = 0$
  \item $f^{(8)}(x) = \cosh x + \cos x \Rightarrow f^{(8)}(0) = 1 + 1 = 2$
\end{enumerate}

$ \Rightarrow f^{(k)}(0) = 0 $ voor $ k = 1, \dots, 7$, en $f^{(8)}(0) = 2 > 0$, dus lokaal minimum in $x = 0$.
