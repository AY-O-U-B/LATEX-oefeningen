$ a(z-i) + b(z+2i) = 3z -i $ \\

$ \Rightarrow az - ai + bz + 2bi = 3z -i $ \\

$\Rightarrow\left\{
	\begin{array}{lll}
		az + bz & = & 3z \\
		-ai + 2bi & = &  -i
	\end{array}
\right . $ \\

$\Rightarrow\left\{
	\begin{array}{lll}
		a + b & = & 3 \\
		-a + 2b & = &  -1
	\end{array}
\right . $ \\

$\Rightarrow\left\{
	\begin{array}{lll}
		a & = & 3 - b \\
		-3 + 3b & = &  -1
	\end{array}
\right . $ \\

$\Rightarrow\left\{
	\begin{array}{lll}
		a & = & 3 - b \\
		3b & = &  2
	\end{array}
\right . $ \\

$\Rightarrow\left\{
	\begin{array}{lll}
		a & = & \dfrac{7}{3} \\
		b & = &  \dfrac{2}{3}
	\end{array}
\right . $ \\
