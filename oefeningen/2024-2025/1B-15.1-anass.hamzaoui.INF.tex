\begin{eqnarray*}
    && \dfrac{1}{\sqrt{2}-i} - \dfrac{1}{\sqrt{2}+i} \\
    && = \dfrac{1}{\sqrt{2}-i} \cdot \dfrac{\sqrt{2}+i}{\sqrt{2}+i} - \dfrac{1}{\sqrt{2}+i} \cdot \dfrac{\sqrt{2}-i}{\sqrt{2}-i} \\
    && = \dfrac{\sqrt{2}+i}{2-i^2} - \dfrac{\sqrt{2}-i}{2-i^2} \\
    && = \dfrac{\sqrt{2}+i}{3} - \dfrac{\sqrt{2}-i}{3} \\
    && = \dfrac{2i}{3} \\
\end{eqnarray*}
