$
\left\{ 
\begin{array}{l} 
f : \mathbb{R}^3 \rightarrow \mathbb{R} : (x,y,z) \rightarrow xyz\\

g : \mathbb{R} \rightarrow \mathbb{R}^2 : x \rightarrow \left(2x, x^3\right)\\

h : \mathbb{R}^2 \rightarrow \mathbb{R}^3 : (x,y) \rightarrow \left(x^2, y^2, x^2y\right)
\end{array} 
\right.
$\\

$f(x,y,z) = xyz \Rightarrow Df(x,y,z) = \left(\begin{array}{c c c} yz & xz & xy \end{array}\right)$\\

$g(x) = (2x, x^3) \Rightarrow Dg(x) = \left(\begin{array}{c} 2 \\ 3x^2 \end{array}\right)$\\

$h(x,y) = (x^2, y^2, x^2y) \Rightarrow Dh(x,y) = \left(\begin{array}{c c} 2x & 0 \\ 0 & 2y \\ 2xy & x^2 \end{array}\right)$\\

Deel 1: $D(h \circ g \circ f)(x,y,z)$\\

$D(h \circ g \circ f)(x,y,z) = Dh(g(f(x,y,z)))   Dg(f(x,y,z))   Df(x,y,z)$\\

$Dh(g(f(x,y,z))) = Dh(2xyz, x^3y^3z^3) = \left(\begin{array}{c c} 4xyz & 0 \\ 0 & 2x^3y^3z^3 \\ 4x^4y^4z^4 & 4x^2y^2z^2 \end{array}\right)$\\

$Dg(f(x,y,z)) = Dg(xyz) = \left(\begin{array}{c} 2 \\ 3x^2y^2z^2 \end{array}\right)$\\

$Df(x,y,z) = \left(\begin{array}{c c c} yz & xz & xy \end{array}\right)$\\

$\Rightarrow D(h \circ g \circ f)(x,y,z) = \left(\begin{array}{c c} 4xyz & 0 \\ 0 & 2x^3y^3z^3 \\ 4x^4y^4z^4 & 4x^2y^2z^2 \end{array}\right)   \left(\begin{array}{c} 2 \\ 3x^2y^2z^2 \end{array}\right)   \left(\begin{array}{c c c} yz & xz & xy \end{array}\right)$\\

$= \left(\begin{array}{c} 8xyz \\ 6x^5y^5z^5 \\ 8x^4y^4z^4 + 12x^4y^4z^4 \end{array}\right)   \left(\begin{array}{c c c} yz & xz & xy \end{array}\right)$\\

$= \left(\begin{array}{c} 8xyz \\ 6x^5y^5z^5 \\ 20x^4y^4z^4 \end{array}\right)   \left(\begin{array}{c c c} yz & xz & xy \end{array}\right)$\\

$= \left(\begin{array}{c c c} 8xy^2z^2 & 8x^2yz^2 & 8x^2y^2z \\ 6x^5y^6z^6 & 6x^6y^5z^6 & 6x^6y^6z^5 \\ 20x^4y^5z^5 & 20x^5y^4z^5 & 20x^5y^5z^4 \end{array}\right)$\\

Deel 2: $D(g \circ f \circ h)(x,y)$\\

$D(g \circ f \circ h)(x,y) = Dg(f(h(x,y)))   Df(h(x,y))   Dh(x,y)$\\

$Dg(f(h(x,y))) = Dg(x^4y^3) = \left(\begin{array}{c} 2 \\ 3x^8y^6 \end{array}\right)$\\

$Df(h(x,y)) = Df(x^2, y^2, x^2y) = \left(\begin{array}{c c c} x^2y^3 & x^4y & x^2y^2 \end{array}\right)$\\

$Dh(x,y) = \left(\begin{array}{c c} 2x & 0 \\ 0 & 2y \\ 2xy & x^2 \end{array}\right)$\\

$\Rightarrow D(g \circ f \circ h)(x,y) = \left(\begin{array}{c} 2 \\ 3x^8y^6 \end{array}\right)   \left(\begin{array}{c c c} x^2y^3 & x^4y & x^2y^2 \end{array}\right)   \left(\begin{array}{c c} 2x & 0 \\ 0 & 2y \\ 2xy & x^2 \end{array}\right)$\\

$= \left(\begin{array}{c c c} 2x^2y^3 & 2x^4y & 2x^2y^2 \\ 3x^{10}y^9 & 3x^{12}y^7 & 3x^{10}y^8 \end{array}\right)   \left(\begin{array}{c c} 2x & 0 \\ 0 & 2y \\ 2xy & x^2 \end{array}\right)$\\

$= \left(\begin{array}{c c} 4x^3y^3 + 4x^3y^3 & 4x^4y^2 + 2x^4y^2 \\ 6x^{11}y^9 + 6x^{11}y^9 & 6x^{12}y^8 + 3x^{12}y^8 \end{array}\right)$\\

$= \left(\begin{array}{c c} 8x^3y^3 & 6x^4y^2 \\ 12x^{11}y^9 & 9x^{12}y^8 \end{array}\right)$\\

Deel 3: $D(f \circ h \circ g)(x)$\\

$D(f \circ h \circ g)(x) = Df(h(g(x)))   Dh(g(x))   Dg(x)$\\

$Df(h(g(x))) = Df(4x^2, x^6, 4x^5) = \left(\begin{array}{c c c} 4x^{11} & 16x^7 & 4x^8 \end{array}\right)$\\

$Dh(g(x)) = Dh(2x, x^3) = \left(\begin{array}{c c} 4x & 0 \\ 0 & 2x^3 \\ 4x^4 & 4x^2 \end{array}\right)$\\

$Dg(x) = \left(\begin{array}{c} 2 \\ 3x^2 \end{array}\right)$\\

$\Rightarrow D(f \circ h \circ g)(x) = \left(\begin{array}{c c c} 4x^{11} & 16x^7 & 4x^8 \end{array}\right)   \left(\begin{array}{c c} 4x & 0 \\ 0 & 2x^3 \\ 4x^4 & 4x^2 \end{array}\right)   \left(\begin{array}{c} 2 \\ 3x^2 \end{array}\right)$\\

$= \left(\begin{array}{c c c} 4x^{11} & 16x^7 & 4x^8 \end{array}\right)   \left(\begin{array}{c} 8x \\ 6x^5 \\ 8x^4 + 12x^4 \end{array}\right)$\\

$= \left(\begin{array}{c c c} 4x^{11} & 16x^7 & 4x^8 \end{array}\right)   \left(\begin{array}{c} 8x \\ 6x^5 \\ 20x^4 \end{array}\right)$\\

$= 32x^{12} + 96x^{12} + 80x^{12}$\\

$= 208x^{12}$\\
