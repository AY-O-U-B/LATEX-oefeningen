$a(3z - 2) + b(z + 3) = 9z + 5$\\

$\Rightarrow 3az - 2a + bz + 3b = 9z + 5$\\

$\Rightarrow\left\{
    \begin{array}{lll}
        3az + bz      & = & 9z,\\
        -2a + 3b      & = & 5
    \end{array}
\right.$\\

$\Rightarrow\left\{
    \begin{array}{lll}
        3a + b        & = & 9,\\
        -2a + 3b      & = & 5
    \end{array}
\right.$\\

$\Rightarrow\left\{
    \begin{array}{lll}
        b             & = & 9 - 3a,\\
        -2a + 3b      & = & 5
    \end{array}
\right.$\\

$\Rightarrow\left\{
    \begin{array}{lll}
        b             & = & 9 - 3a,\\
        -11a          & = & -22
    \end{array}
\right.$\\

$\Rightarrow\left\{
    \begin{array}{lll}
        a             & = & 2,\\
        b             & = & 3
    \end{array}
\right.$
