$f(x)=\dfrac{x^2-5x+6}{x^3-2x^2-x+2} = \dfrac{\left(x-2\right)\left(x-3\right)}{x\left(x^2-1\right)-2\left(x^2-1\right)} = \dfrac{\left(x-2\right)\left(x-3\right)}{\left(x^2-1\right)\left(x-2\right)} = \dfrac{x-3}{x^2-1}$ voor $x\neq2$\\

\begin{itemize}
    \item $\lim\limits_{x\to2} f(x)$\\

    $ \lim\limits_{x\to2}\dfrac{x-3}{x^2-1} = \dfrac{-1}{3}$\\

    \item $\lim\limits_{x\to3} f(x)$\\

    $ \lim\limits_{x\to3}\dfrac{x-3}{x^2-1} = 0$\\

    \item $\lim\limits_{x\to-1} f(x)$\\

    $ \lim\limits_{x\to-1+}\dfrac{x-3}{x^2-1} = \dfrac{x-3}{\left(x-1\right)\left(x+1\right)}=+\infty$\\

    $ \lim\limits_{x\to-1-}\dfrac{x-3}{x^2-1} = \dfrac{x-3}{\left(x-1\right)\left(x+1\right)}=-\infty$\\

    \item $\lim\limits_{x\to1} f(x)$\\

    $ \lim\limits_{x\to1+}\dfrac{x-3}{x^2-1} = \dfrac{x-3}{\left(x-1\right)\left(x+1\right)}=-\infty$\\

    $ \lim\limits_{x\to1-}\dfrac{x-3}{x^2-1} = \dfrac{x-3}{\left(x-1\right)\left(x+1\right)}=+\infty$\\

    \item $\lim\limits_{x\to \infty} f(x)$\\

    $ \lim\limits_{x\to \infty}\dfrac{x-3}{x^2-1} = 0$\\
\end{itemize}
