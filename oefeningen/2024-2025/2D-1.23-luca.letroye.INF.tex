\begin{eqnarray*}
\lim\limits_{x\rightarrow 2}\sqrt{4 - x^2} \Rightarrow && \lim\limits_{x\rightarrow 2-} \sqrt{4-x^2} = \sqrt{4-2^2} = \sqrt{0^+} = 0\\
&& \lim\limits_{x\rightarrow 2+}  \sqrt{4-x^2} = \sqrt{4-2^2} = \sqrt{0^-} = \text{bestaat niet}
\end{eqnarray*}
