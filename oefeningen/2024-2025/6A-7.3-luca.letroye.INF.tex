$x_6 = 1215$\\

$x_{10} = 98415$\\

$\dfrac{x_{10}}{x_6} = r^4 = 81$\\

\noindent Oplossing 1: $r = +3$\\

$r_1 = \sqrt[4]{81} = 3$\\

$x_6 = x_1 \cdot 3^5 = 1215$\\

$x_1= \dfrac{1215}{243} = 5$\\

$x_2 = 5\cdot 3 = 15$\\

$x_3 = 15\cdot 3 = 45$\\

\noindent Oplossing 2: $r = -3$\\

$r_2 = -\sqrt[4]{81} = -3$\\

$x_6 = x_1 \cdot (-3)^5 = x_1 \cdot (-243) = 1215$\\

$x_1= \dfrac{1215}{-243} = -5$\\

$x_2 = -5\cdot (-3) = 15$\\

$x_3 = 15\cdot (-3) = -45$
