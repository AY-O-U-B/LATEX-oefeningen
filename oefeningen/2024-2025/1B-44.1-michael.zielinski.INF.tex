$z^4 + 4(1-i)z^2 + 3 - 4i = 0$ \\
  
$w = z^2$ \\
  
$w^2 + 4(1-i)w + 3 - 4i = 0$ \\
  
$D = 16(1-i)^2 - 4(3-4i)$ \\
  
$(1-i)^2 = -2i$ \\
  
$D = -32i - 12 + 16i = -12 - 16i$ \\
  
$(a+bi)^2 = -12 - 16i$ \\
  
$a^2 - b^2 = -12$, $2ab = -16$ \\
  
$(a,b) = (2,-4)$, $(-2,4)$ \\
  
$\sqrt{D} = \pm (2-4i)$ \\
  
$w = \dfrac{-4(1-i) \pm (2-4i)}{2}$ \\
  
$-4(1-i) = -4 + 4i$ \\
  
$w_1 = \dfrac{(-4 + 4i) + (2 - 4i)}{2} = -1$ \\
  
$w_2 = \dfrac{(-4 + 4i) - (2 - 4i)}{2} = -3 + 4i$ \\
  
$z^2 = -1 \Rightarrow z = \pm i$ \\
  
$z^2 = -3 + 4i$ \\
  
$a^2 - b^2 = -3$, $2ab = 4$ \\
  
$(a,b) = (1,2)$, $(-1,-2)$ \\
  
$z = \pm (1 + 2i)$ \\
  
$\boxed{z \in\{ i, -i, 1+2i, -1-2i\}}$ \\
