$ f(r, s) = \dfrac{r^2 - s^2}{r^2 + s^2} \Rightarrow Df(r,s) = \begin{pmatrix} \dfrac{\partial f}{\partial r} & \dfrac{\partial f}{\partial s} \end{pmatrix} $\\

$ \dfrac{\partial f}{\partial r} = \dfrac{2r\left(r^2+s^2\right) - 2r\left(r^2-s^2\right)}{\left(r^2 + s^2\right)^2} = \dfrac{4rs^2}{\left(r^2 + s^2\right)^2} $\\

$ \dfrac{\partial f}{\partial s} = \dfrac{-2s\left(r^2+s^2\right) - 2s\left(r^2-s^2\right)}{\left(r^2 + s^2\right)^2} = \dfrac{-4sr^2}{\left(r^2 + s^2\right)^2} $\\

$ \Rightarrow Df(r,s) = \left( \begin{array}{cc} \dfrac{4rs^2}{\left(r^2 + s^2\right)^2} & \dfrac{-4sr^2}{\left(r^2 + s^2\right)^2} \end{array} \right)$\\
