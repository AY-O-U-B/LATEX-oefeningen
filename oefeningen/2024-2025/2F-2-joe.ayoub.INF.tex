\begin{eqnarray*}
\cosh(3x) & = & \cosh(x + 2x) \\
& = & \cosh(x) \cdot \cosh(2x) + \sinh(x) \cdot \sinh(2x) \\
& = & \cosh(x) \cdot (\cosh^2(x) + \sinh^2(x)) + \sinh(x) \cdot 2 \cdot \sinh(x) \cdot \cosh(x) \\
& = & \cosh^3(x) + \cosh(x) \cdot \sinh^2(x) + 2 \cdot \sinh^2(x) \cdot \cosh(x) \\
& = & \cosh(x) \cdot (\cosh^2(x) + \sinh^2(x) + 2\sinh^2(x)) \\
& = & \cosh(x) \cdot (\cosh^2(x) + 3\sinh^2(x)) \\
& = & \cosh(x) \cdot (\cosh^2(x) + 3(\cosh^2(x)-1)) \\
& = & \cosh(x) \cdot (\cosh^2(x) + 3\cosh^2(x) - 3) \\
& = & \cosh(x) \cdot (4\cosh^2(x) - 3) \\
\end{eqnarray*}


\begin{eqnarray*}
\sinh(3x) & = & \sinh(x + 2x) \\
& = & \sinh(x) \cdot \cosh(2x) + \cosh(x) \cdot \sinh(2x) \\
& = & \sinh(x) \cdot (\cosh^2(x) + \sinh^2(x)) + \cosh(x) \cdot 2 \cdot \sinh(x) \cdot \cosh(x) \\
& = & \sinh(x) \cdot \cosh^2(x) + \sinh^3(x) + 2\sinh(x)\cosh^2(x) \\
& = & \sinh(x) \cdot (\cosh^2(x) + \sinh^2(x) + 2\cosh^2(x)) \\
& = & \sinh(x) \cdot (\sinh^2(x) + 3\cosh^2(x)) \\
& = & \sinh(x) \cdot (\sinh^2(x) + 3(1 + \sinh^2(x))) \\
& = & \sinh(x) \cdot (\sinh^2(x) + 3 + 3\sinh^2(x)) \\
& = & \sinh(x) \cdot (4\sinh^2(x) + 3) \\
\end{eqnarray*}


\begin{eqnarray*}
\tanh(3x) & = & \tanh(x + 2x) \\
& = & \dfrac{\tanh(x) + \tanh(2x)}{1 + \tanh(x)\tanh(2x)} \\
& = & \dfrac{\tanh(x) + \dfrac{2\tanh(x)}{1 + \tanh^2(x)}}{1 + \tanh(x) \cdot \dfrac{2\tanh(x)}{1 + \tanh^2(x)}} \\
& = & \dfrac{\dfrac{\tanh(x) + \tanh^3(x) + 2\tanh(x)}{1 + \tanh^2(x)}}{\dfrac{1 + \tanh^2(x) + 2\tanh^2(x)}{1 + \tanh^2(x)}} \\
& = & \dfrac{\tanh^3(x) + 3\tanh(x)}{1 + 3\tanh^2(x)} \\
\end{eqnarray*}
