$f(x)=\dfrac{x^2-x-6}{x^2-2x-3} = \dfrac{\left(x-3\right)\left(x+2\right)}{\left(x-3\right)\left(x+1\right)} = \dfrac{x+2}{x+1}$ voor $x\neq3$\\

\begin{itemize}
    \item $\lim\limits_{x\to-1} f(x)$\\

    $ \lim\limits_{x\to-1+}\dfrac{x+2}{x+1} = +\infty$\\

$ \lim\limits_{x\to-1-}\dfrac{x+2}{x+1} = -\infty$\\

    \item $\lim\limits_{x\to-2} f(x)$\\

    $ \lim\limits_{x\to-2}\dfrac{x+2}{x+1} = 0$\\

    \item $\lim\limits_{x\to4} f(x)$\\

    $ \lim\limits_{x\to4}\dfrac{x+2}{x+1} = \dfrac{4+2}{4+1}=\dfrac{6}{5}$\\

    \item $\lim\limits_{x\to3} f(x)$\\

    $ \lim\limits_{x\to3}\dfrac{x+2}{x+1} = \dfrac{3+2}{3+1}=\dfrac{5}{4}$\\

    \item $\lim\limits_{x\to \infty} f(x)$\\

    $ \lim\limits_{x\to \infty}\dfrac{x+2}{x+1} = 1$\\
\end{itemize}
