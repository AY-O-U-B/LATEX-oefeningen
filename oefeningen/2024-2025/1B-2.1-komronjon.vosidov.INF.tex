	\normalsize
	  Bereken de volgende produkten in $\mathbb{C}$:
	
	\begin{equation*}
		(0 + 1i)(a + bi)
	\end{equation*}
	
	
	\noindent \textbf{Oplossing:}
	
	$(0 + 1i)(a + bi) = 0 \cdot a + 0 \cdot bi + 1i \cdot a + 1i \cdot bi$ \\	
	
	$= 0 + 0 + ai + b i^2 $ \\
	
	$= ai - b$ \\
	
	$= -b + ai$ \\
