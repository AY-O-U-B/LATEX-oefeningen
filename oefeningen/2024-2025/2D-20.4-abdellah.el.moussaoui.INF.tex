$\lim\limits_{x \rightarrow \infty} \dfrac{\sqrt{x^2 + 3}}{2x^2 - 1}$\\

$=\lim\limits_{x \rightarrow \infty} \dfrac{\sqrt{x^2\left (1 + \dfrac{3}{x^2}\right )}}{x^2 \left (2 - \dfrac{1}{x^2} \right )}$\\

$=\lim\limits_{x \rightarrow \infty} \dfrac{\left |x \right |\sqrt{1 + \dfrac{3}{x^2}}}{x^2 \left (2 - \dfrac{1}{x^2} \right )}$\\

\medskip

$\lim\limits_{x \rightarrow +\infty} \dfrac{\left |x \right |\sqrt{1 + \dfrac{3}{x^2}}}{x^2 \left (2 - \dfrac{1}{x^2} \right )} = \lim\limits_{x \rightarrow +\infty} \dfrac{x\sqrt{1 + \dfrac{3}{x^2}}}{x^2 \left (2 - \dfrac{1}{x^2} \right )} = 0$\\

$\lim\limits_{x \rightarrow -\infty} \dfrac{\left |x \right |\sqrt{1 + \dfrac{3}{x^2}}}{x^2 \left (2 - \dfrac{1}{x^2} \right )} = \lim\limits_{x \rightarrow -\infty} \dfrac{-x\sqrt{1 + \dfrac{3}{x^2}}}{x^2 \left (2 - \dfrac{1}{x^2} \right )} = 0$\\
