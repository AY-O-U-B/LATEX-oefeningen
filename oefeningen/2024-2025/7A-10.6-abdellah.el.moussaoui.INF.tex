$\dfrac{\partial f}{\partial x} = \dfrac{\dfrac{\partial(xy)}{\partial x} (x^2 + y^2) - (xy) \dfrac{\partial (x^2 + y^2)}{\partial x} }{(x^2 + y^2)^2}$\\

$\Leftrightarrow \dfrac{\partial f}{\partial x} = \dfrac{y (x^2 + y^2) - (xy)(2x)}{(x^2 + y^2)^2}$\\

$\Leftrightarrow \dfrac{\partial f}{\partial x} = \dfrac{y^3 - yx^2}{(x^2 + y^2)^2}$\\

\medskip

$\dfrac{\partial f}{\partial y} = \dfrac{\dfrac{\partial(xy)}{\partial y} (x^2 + y^2) - (xy) \dfrac{\partial (x^2 + y^2)}{\partial y} }{(x^2 + y^2)^2}$\\

$\Leftrightarrow \dfrac{\partial f}{\partial y} = \dfrac{x (x^2 + y^2) - (xy)(2y)}{(x^2 + y^2)^2}$\\

$\Leftrightarrow \dfrac{\partial f}{\partial y} = \dfrac{x^3 - xy^2}{(x^2 + y^2)^2}$\\
