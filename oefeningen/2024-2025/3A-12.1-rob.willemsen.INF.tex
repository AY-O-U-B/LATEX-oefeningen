$f(x) = \left(x^2 - x + 2\right) (3x - 4)$ \\

\begin{enumerate}
    \item Haakjes uitwerken: \\
    
    $= 3x^3 - 3x^2 + 6x - 4x^2 + 4x - 8$ \\

    $= 3x^3 - 7x^2 + 10x - 8$ \\

    $f'(x) = 9x^2 - 14x + 10$ \\
    
    \item Productregel toepassen: \\

    $f'(x) = \left(2x - 1\right) (3x - 4) + \left(x^2 - x + 2\right) \cdot 3$ \\

    $= 6x^2 - 8x - 3x + 4 + 3x^2 - 3x + 6$ \\

    $= 9x^2 - 14x + 10$ \\  
\end{enumerate}
