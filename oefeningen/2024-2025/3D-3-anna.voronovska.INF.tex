$f(x)=ax^3+bx^2$\\

\textbf{Stap 1}\\

$f'(x)= 3ax^2+2bx$\\

$f'(2)=3a \cdot 2^2 + 2b \cdot 2 = 12a+4b$\\

$f'(x)=0$, dus:\\

$12a+4b=0$\\

$3a+b=0$ (*)\\

\textbf{Stap 2}\\

$f(2)=a \cdot 2^3 + b \cdot 2^2=8a+4b$\\

$f(2)=-4$\\

Dus:\\

$8a+4b=-4$\\

$2a+b=-1$ (**)\\

\textbf{Stap 3: Stelsel:} \\

$\left\{
\begin{array}{rcl}
3a + b & = & 0 \\
2a + b & = & -1
\end{array}
\right.$\\

$\left\{
\begin{array}{rcl}
b & = & -3a \\
2a - 3a & = & -1
\end{array}
\right.$\\

$\left\{
\begin{array}{rcl}
b & = & -3a \\
-a & = & -1
\end{array}
\right.$\\

$\left\{
\begin{array}{rcl}
b & = & -3 \cdot 1 \\
a & = & 1
\end{array}
\right.$\\

$\left\{
\begin{array}{rcl}
b & = & -3 \\
a & = & 1
\end{array}
\right.$\\

\textbf{Antwoord:} $a = 1$; $b = -3$.
