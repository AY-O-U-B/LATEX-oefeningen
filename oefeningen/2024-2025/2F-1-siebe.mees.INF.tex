$ \cosh(x - y) = \cosh(x + (-y)) \overset{2.10.4}{=} \cosh x \cosh(-y) + \sinh x \sinh(-y) \overset{\cosh (-x) = \cosh x}{\underset{\sinh(-x) = -\sinh x}{=}} \cosh x \cosh y - \sinh x \sinh y $ \\

$ \sinh(x - y) = \sinh(x + (-y)) \overset{2.10.4}{=} \sinh x \cosh(-y) + \cosh x \sinh(-y) \overset{\cosh (-x) = \cosh x}{\underset{\sinh(-x) = -\sinh x}{=}} \sinh x \cosh y - \cosh x \sinh y $ \\

$ \tanh(x - y) = \tanh(x + (-y)) \overset{2.10.4}{=} \dfrac{\tanh x + \tanh(-y)}{1 + \tanh x \tanh(-y)} \overset{\tanh (-y) = - \tanh y}{=} \dfrac{\tanh x - \tanh y}{1 - \tanh x \tanh y} $ \\
