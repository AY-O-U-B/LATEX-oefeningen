$ 4 \cdot \log_{x}\sqrt3 + \log_{x}3^{\tfrac{1}{3}} + \log_{x}\dfrac{1}{9} + \dfrac{1}{3} = 0 $\\

$ \Leftrightarrow \log_{x}\sqrt3^4 +  \log_{x}3^{\tfrac{1}{3}} + \log_{x}\dfrac{1}{9} = -\dfrac{1}{3} $\\

$ \Leftrightarrow \log_{x}9 +  \log_{x}3^{\tfrac{1}{3}} + \log_{x}\dfrac{1}{9} = -\dfrac{1}{3} $\\

$ \Leftrightarrow \log_{x}\left(9 \cdot \dfrac{1}{9}\right) +  \log_{x}3^{\tfrac{1}{3}} = -\dfrac{1}{3} $\\

$ \Leftrightarrow \log_{x}3^{\tfrac{1}{3}} = -\dfrac{1}{3} $\\

$ \Leftrightarrow x^{-\tfrac{1}{3}} = 3^{\tfrac{1}{3}} $\\

$ \Leftrightarrow x^{-\tfrac{1}{3}} = \left(\dfrac{1}{3}\right)^{-\tfrac{1}{3}} $\\

$ \Leftrightarrow x = \dfrac{1}{3} $\\
