$ r = \cos2\theta $\\

$ \operatorname{dom} = \mathbb{R} $\\

$ p = \pi $\\

$ \rightarrow \operatorname{BD} = [0 , \pi] $\\

$ r = 0 \Leftrightarrow \cos2\theta = 0 \Leftrightarrow2\theta = \dfrac{\pi}{2} + k\pi \Leftrightarrow \theta = \dfrac{\pi}{4} + k\dfrac{\pi}{2},  \forall k \in \mathbb{Z} $\\

$ \rightarrow \left\{ \dfrac{\pi}{4}, \dfrac{3\pi}{4}\right\} $\\

$ r' = 0 \Leftrightarrow -\sin2\theta = 0 \Leftrightarrow 2\theta = k\pi \Leftrightarrow \theta =  k\dfrac{\pi}{2},  \forall k \in \mathbb{Z} $\\

$ \rightarrow \left\{0, \dfrac{\pi}{2}, \pi\right\} $\\
