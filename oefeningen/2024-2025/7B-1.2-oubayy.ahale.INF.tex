    $ w = \dfrac{1}{u^2 + v^2}, u = \cos2t, v = \sin2t $\\

    $ \dfrac{\partial w}{\partial u} = -\dfrac{2u}{(u^2 + v^2)^2}, \dfrac{\partial w}{\partial v} = -\dfrac{2v}{(u^2 + v^2)^2} $\\

    $ \dfrac{du}{dt} = -2\sin2t, \dfrac{dv}{dt} = 2 \cos2t$\\

    $ \dfrac{dw}{dt}= \dfrac{\partial w}{\partial u} \dfrac{du}{dt} + \dfrac{\partial w}{\partial v} \dfrac{dv}{dt} $\\

    $ = \dfrac{4u \sin2t - 4v \cos2t}{(u^2 + v^2)^2} $\\

    $ = 4 \cdot (\cos2t \sin2t - \sin2t \cos2t) = 0$\\
