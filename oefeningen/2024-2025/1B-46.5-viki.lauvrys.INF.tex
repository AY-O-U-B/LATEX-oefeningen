$ \dfrac{z}{z - 2i} - \dfrac{3iz}{(z +i)(z -2i)} = 2 $ \\

$ \Leftrightarrow \dfrac{z (z +i)}{(z - 2i) (z +i)} - \dfrac{3iz}{(z +i)(z -2i)} = 2 $ \\

$ \Leftrightarrow \dfrac{z^2 -2iz}{(z -2i)(z +i)} = 2 $ \\

$ \Leftrightarrow \dfrac{(z -2i)z}{(z -2i)(z +i)} = 2 $ \\

$ \Leftrightarrow \dfrac{z}{z +i} = 2 $ \\

$ \Leftrightarrow z = 2z +2i $ \\

$ \Leftrightarrow 0 = z +2i $ \\

$ \Leftrightarrow z = -2i $ \\
