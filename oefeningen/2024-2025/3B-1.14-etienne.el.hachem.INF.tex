Opgave: $f(x) = \sec \dfrac{1}{x} = \left(cos \dfrac{1}{x}\right)^{-1}$

\begin{eqnarray*}
f'(x) &=& \dfrac{\sin \dfrac{1}{x}}{{\cos \dfrac{1}{x}}^2} \cdot -\dfrac{1}{x^2} \\
			&=& -\dfrac{\tan\dfrac{1}{x}\cdot\sec\dfrac{1}{x}}{x^2} \\
f''(x) &=& \dfrac{(\sec^2\dfrac{1}{x}\cdot\dfrac{1}{x^2}\cdot\sec\dfrac{1}{x}+\tan\dfrac{1}{x}\cdot\dfrac{\tan\dfrac{1}{x}\cdot\sec\dfrac{1}{x}}{x^2})\cdot x^2 - \tan\dfrac{1}{x}\cdot\sec\dfrac{1}{x}\cdot 2x}{x^4} \\
			 &=& \dfrac{\sec^3\dfrac{1}{x}+\tan^2\dfrac{1}{x}\cdot\sec\dfrac{1}{x} - \tan\dfrac{1}{x}\cdot\sec\dfrac{1}{x}\cdot 2x}{x^4} \\
			 &=& \dfrac{\sec\dfrac{1}{x}\left(\sec^2\dfrac{1}{x}+\tan^2\dfrac{1}{x}-\tan\dfrac{1}{x}\cdot 2x\right)}{x^4}
\end{eqnarray*}
