$ T_7(f, 0) (x) $ voor $ f(x) = \tan x $ \\

\begin{enumerate}
  \item $f(x) = \tan x$
  \item $f'(x) = \sec^2 x \quad \Rightarrow \quad f'(0) = 1$
  \item $f''(x) = 2 \sec^2 x \tan x \quad \Rightarrow \quad f''(0) = 0$
  \item $f^{(3)}(x) = 2 \sec^2 x (\sec^2 x + 2 \tan^2 x) \quad \Rightarrow \quad f^{(3)}(0) = 2$
  \item $f^{(4)}(0) = 0$
  \item $f^{(5)}(0) = 16$
  \item $f^{(6)}(0) = 0$
  \item $f^{(7)}(0) = 272$
\end{enumerate}

$T_7(f, 0)(x) = f(0) + \dfrac{f'(0)}{1!}x + \dfrac{f''(0)}{2!}x^2 + \dfrac{f^{(3)}(0)}{3!}x^3 + ... + \dfrac{f^{(7)}(0)}{7!}x^7$ \\

$= 0 + \dfrac{1}{1}x + \dfrac{0}{2}x^2 + \dfrac{2}{6}x^3 + \dfrac{0}{24}x^4 + \dfrac{16}{120}x^5 + \dfrac{0}{720}x^6 + \dfrac{272}{5040}x^7$ \\

$ = x + \dfrac{1}{3}x^3 + \dfrac{2}{15}x^5 + \dfrac{17}{315}x^7$
