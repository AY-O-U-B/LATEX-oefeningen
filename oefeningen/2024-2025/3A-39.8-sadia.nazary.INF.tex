$f(x)=(\ln x)^x$ \\

$f(x)=e^{x\ln(\ln x)}$ \\

$f'(x)=e^{x\ln(\ln x)}(x\ln(\ln x))'$ \\

$=\ln(\ln x)+1/(\ln x)$ \\

$f'(x)=(\ln x)^x(\ln(\ln x)+1/(\ln x))$ \\

$=(\ln x)^x\ln(\ln x)+(\ln x)^x(1/(\ln x))$ \\

$=(\ln x)^x\ln(\ln x)+(\ln x)^{x-1}$ \\

\fbox{$f'(x)=(\ln x)^{x-1}+(\ln x)^x\ln(\ln x)$} \\
