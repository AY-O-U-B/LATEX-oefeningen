		-a+2b &=& -1
	\end{array}
\right .$\\

$\Leftrightarrow\left\{
	\begin{array}{lll} 
		a+b &=& 3\\
		3b &=& -1\\
	\end{array}
\right .$\\

$\Leftrightarrow\left\{
	\begin{array}{lll}

	\end{array}
\right .$\\
