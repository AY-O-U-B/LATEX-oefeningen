Bereken de $n$-de afgeleide van \\

$f(x) = ax^4 + bx^3 + cx^2 +dx + e$  $(a, b, c, d, e \in \mathbb{R}$)\\
\begin{itemize}
	\item $n = 1$\\

$f'(x) = 4ax^3 + 3bx^2 +2cx + d$\\

	\item $n = 2$\\

$f''(x) = 12ax^2 +6bx + 2c$\\

	\item $n = 3$\\
	
$f'''(x) = 24ax + 6b$\\

	\item $n = 4$\\

$f^{iv}(x) = 24a$\\

	\item $n = 5$\\
	
$f^{v}(x) = 0$\\

\end{itemize}

$\Rightarrow n \geq 5$ : $f^{(n)} (x) = 0$
