$ f:\R \rightarrow \R: (x, y) \rightarrow \sqrt{4x^2 + y^2} $ \\

$ \dfrac{\partial f}{\partial x} (x, y) = \dfrac{4x}{\sqrt{4x^2 + y^2}} $ \\

$ \dfrac{\partial f}{\partial y} (x, y) = \dfrac{y}{\sqrt{4x^2 + y^2}} $ \\

$ \dfrac{\partial^2 f}{\partial x^2} (x, y) = \dfrac{4\sqrt{4x^2 + y^2} - \dfrac{16x^2}{\sqrt{4x^2 + y^2}}}{4x^2 + y^2} = \dfrac{4(4x^2+y^2) - 16x^2}{(4x^2+y^2)\sqrt{4x^2+y^2}} = \dfrac{4y^2}{(4x^2+y^2)\sqrt{4x^2+y^2}} $ \\

$ \dfrac{\partial^2 f}{\partial xy} (x, y) = \dfrac{-4xy}{(4x^2+y^2)\sqrt{4x^2+y^2}} $ \\

$ \dfrac{\partial^2 f}{\partial yx} (x, y) = \dfrac{-4xy}{(4x^2+y^2)\sqrt{4x^2+y^2}} $ \\

$ \dfrac{\partial^2 f}{\partial y^2} (x, y) = \dfrac{\sqrt{4x^2+y^2} - \dfrac{y^2}{\sqrt{4x^2+y^2}}}{4x^2+y^2} = \dfrac{4x^2}{(4x^2+y^2)\sqrt{4x^2+y^2}} $ \\

$ D^2 f = \left( \begin{array}{cc}
    \dfrac{4y^2}{(4x^2+y^2)\sqrt{4x^2+y^2}} & \dfrac{-4xy}{(4x^2+y^2)\sqrt{4x^2+y^2}} \\
    \dfrac{-4xy}{(4x^2+y^2)\sqrt{4x^2+y^2}} & \dfrac{4x^2}{(4x^2+y^2)\sqrt{4x^2+y^2}} \\
\end{array} \right) $
