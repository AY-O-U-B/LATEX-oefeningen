$f(x) = \dfrac{1}{2}\lfloor x \rfloor - \dfrac{1}{2}x$\\

\begin{tabular}{l l}
$\lim\limits_{x \to 0+} f(x) = 0$ & $\lim\limits_{x \to 0-} f(x) = \dfrac{-1}{2}$\\

$\lim\limits_{x \to 4+} f(x) = 0$  & $\lim\limits_{x \to 4-} f(x) =\dfrac{-1}{2}$\\

$\lim\limits_{x \to \dfrac{1}{2}} f(x) = \dfrac{-1}{4}$\\

$\lim\limits_{x \to \sqrt{2}} f(x) = \dfrac{1-\sqrt{2}}{2}$\\

$\lim\limits_{x \to \dfrac{-5}{4}} f(x) = \dfrac{-3}{8}$\\

$\lim\limits_{x \to \infty} f(x)$ bestaat niet\\
\end{tabular}
