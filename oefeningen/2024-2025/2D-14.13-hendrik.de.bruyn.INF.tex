\textbf{Opgave:} Bereken de limiet $\lim\limits_{x \rightarrow 0} \dfrac{2(1-\cos x)}{x^2}$. \\

\textbf{Oplossing:} \\

$\begin{array}{rcl}
\lim\limits_{x \rightarrow 0} \dfrac{2(1-\cos x)}{x^2}
&\overset{x = 2y}{=}& \lim\limits_{y \rightarrow 0} \dfrac{2(1-\cos 2y)}{4y^2} \\
&=& \lim\limits_{y \rightarrow 0} \dfrac{4\dfrac{1-\cos 2y}{2}}{4y^2} \\
&=& \lim\limits_{y \rightarrow 0} \dfrac{4\sin^2 y}{4y^2} \\
&=& \lim\limits_{y \rightarrow 0} \dfrac{\sin^2 y}{y^2} \\
&=& \left(\lim\limits_{y \rightarrow 0} \dfrac{\sin y}{y}\right)^2 \\
&=& (1)^2 \\
&=& 1 \\
\end{array}
$
