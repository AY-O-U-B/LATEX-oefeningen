\textbf{Opgave:} Bereken de limiet $\lim\limits_{x \rightarrow 2} \dfrac{x^3 - x^2 - 8x + 12}{3x^3 - 10x^2 +4x + 8}$ \\

\textbf{Oplossing:} \\

\begin{tabular}{c|ccccccc}
    & $1$ && $-1$ && $-8$ && $12$ \\
    $2$ & && $2$ && $2$ && $-12$ \\ \hline
    & $1$ && $1$ && $-6$ &\vline&$0$ \\
    $2$ & && $2$ && $6$ \\ \cline{0-5}
    & $1$ && $3$ &\vline& $0$ \\
\end{tabular} \\

$\Rightarrow x^3 - x^2 - 8x + 12 = (x-2)^2(x+3)$ \\

\begin{tabular}{c|ccccccc}
    & $3$ && $-10$ && $4$ && $8$ \\
    $2$ & && $6$ && $-8$ && $-8$ \\ \hline
    & $3$ && $-4$ && $-4$ &\vline&$0$ \\
    $2$ & && $6$ && $4$ \\ \cline{0-5}
    & $3$ && $2$ &\vline& $0$ \\
\end{tabular} \\

$\Rightarrow 3x^3 - 10x^2 +4x + 8 = (x-2)^2(3x+2)$ \\

$\Rightarrow \lim\limits_{x \rightarrow 2} \dfrac{(x-2)^2(x+3)}{(x-2)^2(3x+2)}
= \lim\limits_{x \rightarrow 2} \dfrac{x+3}{3x+2}
= \dfrac{5}{8}$ \\
