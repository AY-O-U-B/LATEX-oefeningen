$f(x) = \dfrac{1}{\sqrt{x}} = x^{- \tfrac{1}{2}}$\\

$f'(x) = - \dfrac{1}{2}x^{- \tfrac{3}{2}}$\\

$f''(x) = \left(- \dfrac{1}{2} \cdot \left( - \dfrac{3}{2}\right)\right)x^{- \tfrac{5}{2}} = \dfrac{3}{4}x^{- \tfrac{5}{2}}$\\

$f'''(x) = \left( \dfrac{3}{4} \cdot \left( - \dfrac{5}{2}\right)\right)x^{- \tfrac{7}{2}} = - \dfrac{15}{8}x^{- \tfrac{7}{2}}$\\

$f^{iv}(x) = \left(- \dfrac{15}{8} \cdot \left( - \dfrac{7}{2} \right)\right)x^{- \tfrac{9}{2}} = \dfrac{105}{16}x^{- \tfrac{9}{2}}$\\
