\noindent \textbf{Schets de functie:} $r = 1 + \cos(\theta)$\\

$\left .
\begin{array}{l}
    \operatorname{dom} r    = \mathbb{R}\\
    P                       = 2\pi
\end{array}
\right \}
\text{Bep. Dom.} = [0, 2\pi]$\\

\noindent \textbf{Nulpunten:}\\

$r = 0 \Leftrightarrow \cos(\theta) = -1 \Leftrightarrow \theta = \pi \longrightarrow \theta \in \{\pi\}$\\

$r' = 0 \Leftrightarrow -\sin(\theta) = 0 \Leftrightarrow \theta = k\pi$ met $k \in \mathbb{N} \longrightarrow \theta \in \{ 0, \pi, 2\pi\}$\\
