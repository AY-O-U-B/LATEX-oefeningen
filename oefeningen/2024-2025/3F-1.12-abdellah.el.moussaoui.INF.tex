$f(x) = \dfrac{x^3+4x^2+x+1}{x-3}$\\

\textbf{Verticale Asymptoot:}\\

$x-3 = 0 \Leftrightarrow x = 3$\\

$\lim\limits_{x \rightarrow 3+} \dfrac{x^3+4x^2+x+1}{x-3} = +\infty$\\

$\lim\limits_{x \rightarrow 3-} \dfrac{x^3+4x^2+x+1}{x-3} = -\infty$\\

Verticale asymptoot op $x = 3$\\

$\begin{array}{r|c c c c}
x & & 3 & &\\ 
\hline f(x) & + & | & - &
\end{array}$ \\

\medskip

\textbf{Horizontale Asymptoot:}\\

$\lim\limits_{x \rightarrow \infty} \dfrac{x^3+4x^2+x+1}{x-3} = \lim\limits_{x \rightarrow \infty} \dfrac{x^3}{x} = \lim\limits_{x \rightarrow \infty} x^2 = +\infty$\\

Geen HA\\

\medskip

\textbf{Schuine Asymptoot:}\\

$\lim\limits_{x \rightarrow \infty} \dfrac{\dfrac{x^3+4x^2+x+1}{x-3}}{x} = \lim\limits_{x \rightarrow \infty} \dfrac{x^3+4x^2+x+1}{x^2-3x} = \lim\limits_{x \rightarrow \infty} \dfrac{x^3}{x^2} = \lim\limits_{x \rightarrow \infty}  x = \infty$\\

Geen SA\\
