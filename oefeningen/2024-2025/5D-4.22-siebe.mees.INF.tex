$y_1 = 2 - \sqrt{x} $ en  $ y_2 = \dfrac{\sqrt{x} + 1}{2\sqrt{x}}$\\

\begin{enumerate}

\item Snijpunten \\

$2 - \sqrt{x} = \dfrac{\sqrt{x} + 1}{2\sqrt{x}}$\\

$2\sqrt{x}(2 - \sqrt{x}) = \sqrt{x} + 1$\\

$4\sqrt{x} - 2x = \sqrt{x} + 1$\\

$3\sqrt{x} - 2x - 1 = 0$\\

$t = \sqrt{x}$\\

$3t - 2t^2 - 1 = 0 \Longleftrightarrow -2t^2 + 3t - 1 = 0$\\

$t = \dfrac{-3 \pm \sqrt{9 - 4\cdot(-2)\cdot(-1)}}{-4}$\\

$t = \dfrac{-3 \pm \sqrt{1}}{-4}$\\

$t = 1 $ of $ t = \dfrac{1}{2}$\\

$\sqrt{x} = 1 \Rightarrow x = 1$\\

$\sqrt{x} = \dfrac{1}{2} \Rightarrow x = \dfrac{1}{4}$\\

\item Bepaal bovenste functie\\

Neem $ x = \dfrac{1}{2} $\\

$y_1 = 2 - \sqrt{\tfrac{1}{2}} \approx 1,29$\\

$y_2 = \dfrac{\sqrt{\tfrac{1}{2}} + 1}{2\sqrt{\tfrac{1}{2}}} \approx 1,21$\\

$\Rightarrow y_1 $ boven $  y_2$\\

\end{enumerate}

$\displaystyle\int_{\tfrac{1}{4}}^{1} \left(2 - \sqrt{x} - \dfrac{\sqrt{x} + 1}{2\sqrt{x}} \right) dx$\\

$= \displaystyle\int_{\tfrac{1}{4}}^{1} \left(2 - \sqrt{x} - \dfrac{1}{2} - \dfrac{1}{2\sqrt{x}}\right) dx$\\

$= \displaystyle\int_{\tfrac{1}{4}}^{1} \left( \dfrac{3}{2} - \sqrt{x} - \dfrac{1}{2\sqrt{x}} \right) dx$\\

$= \left[ \dfrac{3}{2}x - \dfrac{2}{3}x^{\tfrac{3}{2}} - \sqrt{x} \right]_{\tfrac{1}{4}}^{1}$\\

$= \left( -\dfrac{3}{2} + \dfrac{2}{3} + 1 \right) - \left( -\dfrac{3}{8} + \dfrac{2}{3}\cdot\dfrac{1}{8} + \dfrac{1}{2} \right)$\\

$= \left( -\dfrac{1}{6} \right) - \left( -\dfrac{5}{24} \right)$\\

$= \dfrac{1}{24}$\\
