\begin{eqnarray*}
    && \dfrac{1 + 3i}{3 - i} \\
    && = \dfrac{(1 + 3i)(3 + i)}{(3 - i)(3 + i)} \\
    && = \dfrac{3 + i + 9i + 3i^2}{9 - i^2} \\
    && = \dfrac{3 + 10i - 3}{9 + 1} \\
    && = \dfrac{10i}{10} \\
    && = i \\
\end{eqnarray*}
