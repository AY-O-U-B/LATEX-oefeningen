$\tanh^2 x + \sech^2 x = 1$  \\

$\Leftrightarrow \dfrac{\sinh^2 x}{\cosh^2 x} + \dfrac{1}{\cosh^2 x} = 1$  \\

$\Leftrightarrow \left( \dfrac{\frac{e^x - e^{-x}}{2}}{\frac{e^x + e^{-x}}{2}} \right)^2 + \dfrac{1}{\cosh^2 x} = 1$  \\

$\Leftrightarrow \dfrac{(e^x - e^{-x})^2}{(e^x + e^{-x})^2} + \dfrac{4}{(e^x + e^{-x})^2} = 1$  \\

$\Leftrightarrow \dfrac{e^{2x} - 2 + e^{-2x}}{e^{2x} + 2 + e^{-2x}} + \dfrac{4}{e^{2x} + 2 + e^{-2x}} = 1$  \\

$\Leftrightarrow \dfrac{e^{2x} - 2 + e^{-2x} + 4}{e^{2x} + 2 + e^{-2x}} = 1$  \\

$\Leftrightarrow \dfrac{e^{2x} + 2 + e^{-2x}}{e^{2x} + 2 + e^{-2x}} = 1$  \\

$\Leftrightarrow 1 = 1$  \\
