$ f : \mathbb{R}^3 \rightarrow \mathbb{R} : (x,y,z) \rightarrow xyz $\\

$ g : \mathbb{R} \rightarrow \mathbb{R}^2 : x \rightarrow \left(2x, x^3\right) $\\

$ h : \mathbb{R}^2 \rightarrow \mathbb{R}^3 : (x,y) \rightarrow \left(x^2, y^2, x^2y\right) $\\

$ Dh = \left(\begin{array}{c c} 2x & 0 \\ 0 & 2y \\ 2xy & x^2 \end{array} \right) $\\

$ Dg = \left(\begin{array}{c} 2 \\ 3x^2 \end{array}\right) $\\

$ Df = \left(\begin{array}{c c c} yz & xz & xy \end{array}\right) $\\

$ D(h \circ g \circ f) = Dh(g(f(x))) \cdot Dg(f(x)) \cdot Df(x) $\\

$ Df(x) = \left(\begin{array}{c c c} yz & xz & xy \end{array}\right) $\\

$ Dg(f(x)) = \left(\begin{array}{c} 2 \\ 3(xyz)^2 \end{array}\right) $\\

$ Dh(g(f(x))) = \left(\begin{array}{c c} 4xyz & 0 \\ 0 & 2(xyz)^3 \\ 4(xyz)^4 & 4(xyz)^2 \end{array} \right) $\\

$ \Rightarrow D(h \circ g \circ f)(x) = \left(\begin{array}{c c} 4xyz & 0 \\ 0 & 2(xyz)^3 \\ 4(xyz)^4 & 4(xyz)^2 \end{array} \right)
 \cdot \left(\begin{array}{c} 2 \\ 3(xyz)^2 \end{array}\right) \cdot  \left(\begin{array}{c c c} yz & xz & xy \end{array}\right) $\\

$ = \left(\begin{array}{c} 8xyz \\ 6(xyz)^5 \\ 20(xyz)^4 \end{array}\right) \cdot \left(\begin{array}{c c c} yz & xz & xy \end{array}\right) $\\

$ = \left(\begin{array}{c c c} 8xy^2z^2 & 8x^2yz^2 & 8x^2y^2z \\ 6x^5y^6z^6 & 6x^6y^5z^6 & 6 x^6y^6z^5 
\\ 20x^4y^5z^5 & 20 x^5y^4z^5 & 20x^5y^5z^4 \end{array}\right) $\\

$ D(g \circ f \circ h)(x) = Dg(f(g(x))) \cdot Df(h(x)) \cdot Dh(x) $\\

$ Dh(x) =  \left(\begin{array}{c c} 2x & 0 \\ 0 & 2y \\ 2xy & x^2 \end{array} \right) $\\

$ Df(h(x)) =  \left(\begin{array}{c c c} x^2y^3 & x^4y & x^2y^2 \end{array}\right) $\\

$ Dg(f(g(x))) = \left(\begin{array}{c} 2 \\ 3(x^4y^3)^2 \end{array}\right) $\\

$ \Rightarrow D(g \circ f \circ h) = \left(\begin{array}{c} 2 \\ 3(x^4y^3)^2 \end{array}\right) \cdot \left(\begin{array}{c c c} x^2y^3 & x^4y & x^2y^2 \end{array}\right)
\cdot \left(\begin{array}{c c} 2x & 0 \\ 0 & 2y \\ 2xy & x^2 \end{array} \right) $\\

$ = \left(\begin{array}{c c c} 2x^2y^3 & 2x^4y & 2x^2y^2 \\ 3x^{10}y^9 & 3x^12y^7 & 3x^{10}y^8 \end{array}\right) 
\cdot \left(\begin{array}{c c} 2x & 0 \\ 0 & 2y \\ 2xy & x^2 \end{array} \right) $\\

$ = \left(\begin{array}{c c} 8x^3y^3 & 6x^4y^2 \\ 12x^{11}y^9 & 9x^{12}y^8 \end{array}\right) $\\

$ D(f \circ h \circ g)(x) = Df(h(g(x))) \cdot Dh(g(x)) \cdot Dg(x) $\\

$ Dg(x) = \left(\begin{array}{c} 2 \\ 3x^2 \end{array}\right) $\\

$ Dh(g(x)) =  \left(\begin{array}{c c} 4x & 0 \\ 0 & 2x^3 \\ 4x^4 & 4x^2 \end{array} \right) $\\

$ Df(h(g(x))) = \left(\begin{array}{c c c} 4x^{11} & 16x^7 & 4x^8 \end{array}\right) $\\

$ \Rightarrow D(f \circ h \circ g)(x) = \left(\begin{array}{c c c} 4x^{11} & 16x^7 & 4x^8 \end{array}\right)
\cdot \left(\begin{array}{c c} 4x & 0 \\ 0 & 2x^3 \\ 4x^4 & 4x^2 \end{array} \right) \cdot \left(\begin{array}{c} 2 \\ 3x^2 \end{array}\right) $\\

$ = \left(\begin{array}{c c c} 4x^{11} & 16x^7 & 4x^8 \end{array}\right) \cdot \left(\begin{array}{c} 2 \\ 3x^2 \end{array}\right) 
= 64x^{12} + 144x^{12} = 208x^{12} $\\
