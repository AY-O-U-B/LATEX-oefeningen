$f(x) = \sin x + \dfrac{1}{2}\sin 2x$\\

\begin{enumerate}
	\item Domein: $\mathbb{R}$\\

	\item Asymptoten: geen, periodieke functie\\

	\item Tekenonderzoek $f(x)$\\

	$f(x) = \sin x + \sin x\cos x = \sin x(1+\cos x)$\\

	$f(x) = 0 \Leftrightarrow \sin x = 0 \text{ of } \cos x = -1$\\

	$ x = n\pi \text{ en } x = \pi + 2k\pi$\\

	$f^{-1}(0) = {x \in \mathbb{R} | x = n\pi, n \in \mathbb{Z}}$\\

	\item Tekenonderzoek $f'(x)$\\

	$f(x) = \cos x + \dfrac{1}{2}\cdot 2\cos 2x = \cos x + \cos 2x = 2\cos\left(\dfrac{3x}{2}\right)\cos\left(\dfrac{x}{2}\right)$\\

	$f'(x) = 0 \Leftrightarrow \cos\left(\dfrac{3x}{2}\right) = 0 \text{ of } \cos\left(\dfrac{x}{2}\right) = 0$\\

	$ x = \pm \dfrac{\pi}{3} + 2k\pi \text{ en } x = \pi+2k\pi$\\

	\item Tekenonderzoek $f''(x)$\\

	$f''(x) = -\sin x - 2\sin 2x$\\

	$ = -\sin x - 4\sin x\cos x = \sin x(-1-4\cos x) = 0$\\

	$\sin x = 0 \Rightarrow x = k\pi$\\

	$\cos x = -\dfrac{1}{4} \Rightarrow \pm \operatorname{Bgcos}\left(-\dfrac{1}{4}\right)+2k\pi$\\
\end{enumerate}
