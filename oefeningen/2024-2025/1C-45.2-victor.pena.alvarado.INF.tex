	$z^3 = -1 = -1 + 0i \Rightarrow
	\left\{
	\begin{array}{l}
		x=-1\\
		y = 0
	\end{array}
	\right .$\\
	
	$r = \sqrt{(-1)^2 + 0^2} = 1$ \\
	
	$y=0, x<0 \Rightarrow \theta = \pi$
	
	\begin{eqnarray*}
		w_0 & = & \sqrt[3]{1}\cis\left(\dfrac{\pi + (0)2\pi}{3}\right) \\
		& = & \cis\left(\dfrac{\pi}{3}\right) \\
		& = & \cos\dfrac{\pi}{3} + i\sin\dfrac{\pi}{3} \\
		& = & \dfrac{1}{2} + \dfrac{\sqrt{3}}{2}i
	\end{eqnarray*}
	\begin{eqnarray*}
		w_1 & = & \sqrt[3]{1}\cis\left(\dfrac{\pi + (1)2\pi}{3}\right) \\
		& = & \cis\left(\pi\right) \\
		& = & \cos\pi + i\sin\pi \\
		& = & -1 + 0i \\
		& = & -1
	\end{eqnarray*}
	\begin{eqnarray*}
		w_2 & = & \sqrt[3]{1}\cis\left(\dfrac{\pi + (2)2\pi}{3}\right) \\
		& = & \cis\left(\dfrac{5\pi}{3}\right) \\
		& = & \cos\dfrac{5\pi}{3} + i\sin\dfrac{5\pi}{3} \\
		& = & \dfrac{1}{2} - \dfrac{\sqrt{3}}{2}i
	\end{eqnarray*}
