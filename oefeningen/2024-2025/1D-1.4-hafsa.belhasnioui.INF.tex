$a(2zi - 3 + i) + b(z + 1 + i) = 5$ \\

$\Leftrightarrow 2aiz - 3a + ai + bz + b + bi = 5$ \\

$\Leftrightarrow z(2ai + b) - 3a + ai + b + bi = 5$ \\

$\Rightarrow
\left \{
\begin{array}{l}
2ai + b = 0 \\
-3a + ai + b + bi = 5\\
\end{array}
\right.$ \\

$\Rightarrow \left \{
\begin{array}{l}
b = -2ai\\
-3a + ai - 2ai + i(-2ai)= 5\\
\end{array}
\right.$ \\

$\Rightarrow \left \{
\begin{array}{l}
b = -2ai\\
-3a - ai + 2a = 5\\
\end{array}
\right.$ \\

$\Rightarrow \left \{
\begin{array}{l}
b = -2ai\\
-a - ai = 5\\
\end{array}
\right.$ \\

$\Rightarrow \left \{
\begin{array}{l}
b = -2ai\\
a = \dfrac{5}{-1-i} \cdot \dfrac{(-1 + i)}{(-1 + i)}\\
\end{array}
\right.$ \\

$\Rightarrow \left \{
\begin{array}{l}
b = -2ai\\
a = -\dfrac{5}{2} + \dfrac{5}{2}i\\
\end{array}
\right. $\\

$\Rightarrow \left \{
\begin{array}{l}
b = -2\left(-\dfrac{5}{2} + \dfrac{5}{2}i\right)i\\
a = -\dfrac{5}{2} + \dfrac{5}{2}i\\
\end{array}
\right.$ \\

$\Rightarrow \left \{
\begin{array}{l}
b = 5i + 5\\
a = -\dfrac{5}{2} + \dfrac{5}{2}i\\
\end{array}
\right.$ \\
