$z^5+(3-2i)z^4+2(1-3i)z^3-2(1+3i)z^2-(3+2i)z-1 = 0$ \\

$\Rightarrow z^5+(3-2i)z^4+(2-6i)z^3+(-2-6i)z^2+(-3-2i)z-1 = 0$ \\

\begin{tabular}{c|ccccccc}
  & $1$ & $3-2i$ & $2-6i$ & $-2-6i$ & $-3-2i$ & \vline & $-1$ \\
$i$ & & $i$ & $1+3i$ & $3+3i$ & $3+i$ & \vline & $1$ \\ \hline
  & $1$ & $3-i$ & $3-3i$ & $1-3i$ & $-i$ & \vline & $0$
\end{tabular} \\

$\Rightarrow (z-i)(z^4+(3-i)z^3+(3-3i)z^2+(1-3i)z-i) = 0$ \\

\begin{tabular}{r|cccccc}
  & $1$ & $3-i$ & $3-3i$ & $1-3i$ & \vline & $-i$ \\
$-1$ & & $-1$ & $-2+i$ & $-1+2i$ & \vline & $i$ \\ \hline
  & $1$ & $2-i$ & $1-2i$ & $-i$ & \vline & $0$
\end{tabular} \\

$\Rightarrow (z-i)(z+1)(z^3+(2-i)z^2+(1-2i)z-i) = 0$ \\

\begin{tabular}{r|ccccc}
  & $1$ & $2-i$ & $1-2i$ & \vline & $-i$ \\
$-1$ & & $-1$ & $-1+i$ & \vline & $i$ \\ \hline
  & $1$ & $1-i$ & $-i$ & \vline & $0$
\end{tabular} \\

$\Rightarrow (z-i)(z+1)^2(z^2+(1-i)z-i) = 0$ \\

\begin{tabular}{r|cccc}
  & $1$ & $1-i$ & \vline & $-i$ \\
$-1$ & & $-1$ & \vline & $i$ \\ \hline
  & $1$ & $-i$ & \vline & $0$
\end{tabular} \\

$\Rightarrow (z-i)(z+1)^3(z-i) = 0$ \\

$\Rightarrow (z+1)^3(z-i)^2 = 0$ \\

$z_1 = i$ \\
