$f(x,y) = \sqrt{x^2 + y^2} \\$

$f(3,-4) = \sqrt{3^2 + (-4)^2} = \sqrt{9 + 16} = 5 \\$

$\dfrac{\partial f}{\partial x}(x,y) = \dfrac{\partial}{\partial x} \sqrt{x^2 + y^2} = \dfrac{2x}{2\sqrt{x^2 + y^2}} = \dfrac{x}{\sqrt{x^2 + y^2}} \\$

$\dfrac{\partial f}{\partial y}(x,y) = \dfrac{\partial}{\partial y} \sqrt{x^2 + y^2} = \dfrac{2y}{2\sqrt{x^2 + y^2}} = \dfrac{y}{\sqrt{x^2 + y^2}} \\$

$\dfrac{\partial f}{\partial x}(3,-4) = \dfrac{3}{\sqrt{3^2 + (-4)^2}} = \dfrac{3}{5} \\$

$\dfrac{\partial f}{\partial y}(3,-4) = \dfrac{-4}{\sqrt{3^2 + (-4)^2}} = \dfrac{-4}{5} \\$

$z - f(3,-4) = \dfrac{\partial f}{\partial x}(3,-4)(x - 3) + \dfrac{\partial f}{\partial y}(3,-4)(y - (-4)) \\$

$z - 5 = \dfrac{3}{5}(x - 3) - \dfrac{4}{5}(y + 4) \\$

$z = 5 + \dfrac{3}{5}x - \dfrac{9}{5} - \dfrac{4}{5}y - \dfrac{16}{5} \\$

$z = \dfrac{3}{5}x - \dfrac{4}{5}y + \dfrac{25}{5} - \dfrac{9}{5} - \dfrac{16}{5} \\$

$z = \dfrac{3}{5}x - \dfrac{4}{5}y \\$
