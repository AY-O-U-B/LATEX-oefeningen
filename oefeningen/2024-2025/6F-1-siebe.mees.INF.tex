$ f_n:\mathbb{R}\rightarrow \mathbb{R}: x\mapsto \sin(nx) $\\

\begin{enumerate}

\item Puntsgewijze convergentie

$ x=\dfrac{\pi}{2}: f_n\left(\dfrac{\pi}{2}\right)=\sin\left(\dfrac{n\pi}{2}\right) $\\

$ \left(f_n\left(\dfrac{\pi}{2}\right)\right)_n=(1,0,-1,0,1,\ldots) $\\

$ \lim\limits_{n \rightarrow \infty} f_n\left(\dfrac{\pi}{2}\right) $ bestaat niet $ $\\

$ \Rightarrow (f_n)_n $ convergeert niet puntsgewijs op $ \mathbb{R} $\\

\item Uniforme convergentie

$ (f_n)_n $ convergeert niet puntsgewijs op $ \mathbb{R} $\\

$ \Rightarrow (f_n)_n $ convergeert ook niet uniform op $ \mathbb{R} $\\

\end{enumerate}
