\textbf{Opgave:} \\

Bepaal het quotiënt en de rest van: \\

\[
(3z^3 - 4z^2 + 6z - 8) \div (z - 1)
\]

\textbf{Stap 1: Horner-schema (met $z = 1$):}

\[
\begin{array}{r|rrrr}
 & 3 & -4 & 6 & -8 \\
1 &   & 3 & -1 & 5 \\ \hline
 & 3 & -1 & 5 & -3
\end{array}
\]

\textbf{Stap 2: Interpretatie van het resultaat:}\\

De onderste rij geeft:

\begin{itemize}
  \item Coëfficiënten van het quotiënt: $3z^2 - z + 5$
  \item Rest: $-3$
\end{itemize}

\textbf{Conclusie:}
\[
\boxed{
(3z^3 - 4z^2 + 6z - 8) = (z - 1)(3z^2 - z + 5) + (-3)
}
\]
