$ \lim\limits_{x \rightarrow 7} \left(\dfrac{3 - \sqrt{x + 2}}{x - 7} \right) 
= \lim\limits_{x \rightarrow 7} \left(\dfrac{(3 - \sqrt{x + 2})(3 + \sqrt{x + 2})}{(x - 7)(3 + \sqrt{x + 2})} \right) 
= \lim\limits_{x \rightarrow 7} \left(\dfrac{9 - x - 2}{(x - 7)(3 + \sqrt{x + 2})} \right) $\\

\medskip

$ = \lim\limits_{x \rightarrow 7} \left(\dfrac{-(x - 7)}{(x - 7)(3 + \sqrt{x + 2})} \right) 
= \lim\limits_{x \rightarrow 7} \left(\dfrac{-1}{3 + \sqrt{x + 2}} \right) =\dfrac{-1}{3 + \sqrt{9}} = -\dfrac{1}{6} $\\
