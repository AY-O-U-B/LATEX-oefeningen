\begin{tabular}{c|ccc|c}
    & $1$ & $4+2i$ & $5+6i$ & $2+4i$ \\
    $-1$ & & $-1$ & $-3-2i$ & $-2-4i$ \\ \hline
    & $1$ & $3+2i$ & $2+4i$ & $0$
\end{tabular} \\ 

$ \Leftrightarrow (z + 1)(z^2 + (3+2i)z + 2 + 4i) = 0 $ \\

$ \Delta = (3+2i)^2 - 4(2+4i) = (9 + 12i - 4) -(8+16i) = -3-4i $ \\

$\Leftrightarrow \left\{
\begin{array}{l}
    x^2 - y^2 = -3 \\
    2xy = -4 \\
\end{array} \right. $ \\

$\Leftrightarrow \left\{
\begin{array}{l}
    x^2 - y^2 = -3 \\
    x^2(-y)^2 = 4 \\
\end{array} \right. $ \\

$ \lambda^2 + 3\lambda - 4 = 0 $ \\

$ \Delta_\lambda = 3^2 - 4*(-4) = 25 $ \\

$ \Leftrightarrow \lambda = \dfrac{-3 \pm \sqrt{25}}{2} = \dfrac{-3 \pm 5}{2} $ \\

$ \Leftrightarrow \lambda_1 = 1 $ en $ \lambda_2 = -4 $ \\

$ \Leftrightarrow x = \pm 1 $ en $ y = \pm 2 $ \\

$ \Leftrightarrow k_1 = 1 - 2i $ en $ k_2 = -1 + 2i $ \\

$ \Leftrightarrow z_3 = \dfrac{-3 -2i + 1 -2i}{2} = -1 - 2i $ en $ z_4 = \dfrac{-3-2i -1+2i}{2} = -2 $ \\
