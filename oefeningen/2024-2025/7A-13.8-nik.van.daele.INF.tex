$f(x, y, z) = \sqrt{x^2 + y^2 + z^2}$\\

$\dfrac{\partial f}{\partial x} = \dfrac{1}{2} \dfrac{1}{\sqrt{x^2 + y^2 + z^2}} 2x = \dfrac{x}{\sqrt{x^2 + y^2 + z^2}}$\\

$\dfrac{\partial f}{\partial y} = \dfrac{1}{2} \dfrac{1}{\sqrt{x^2 + y^2 + z^2}} 2y = \dfrac{y}{\sqrt{x^2 + y^2 + z^2}}$\\

$\dfrac{\partial f}{\partial z} = \dfrac{1}{2} \dfrac{1}{\sqrt{x^2 + y^2 + z^2}} 2z = \dfrac{z}{\sqrt{x^2 + y^2 + z^2}}$\\

$\nabla f (x, y, z) = \left(\dfrac{x}{\sqrt{x^2 + y^2 + z^2}}, \dfrac{y}{\sqrt{x^2 + y^2 + z^2}}, \dfrac{z}{\sqrt{x^2 + y^2 + z^2}}\right)$\\

$\nabla f (12, 3, 4) = \left(\dfrac{12}{13}, \dfrac{3}{13}, \dfrac{4}{13}\right)$\\
