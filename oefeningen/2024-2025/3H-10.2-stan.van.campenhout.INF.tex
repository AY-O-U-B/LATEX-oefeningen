$ r = \sin\dfrac{\theta}{2} $\\

$ \operatorname{dom} = \mathbb{R} $\\

$ p = 4\pi $\\

$ \rightarrow \operatorname{BD} = [0 , 4\pi] $\\

$ r = 0 \Leftrightarrow \sin\dfrac{\theta}{2} = 0 \Leftrightarrow \dfrac{\theta}{2} = k\pi \Leftrightarrow \theta = 2k\pi,  \forall k \in \mathbb{Z} $\\

$ \rightarrow \left\{0, 2\pi, 4\pi\right\} $\\

$ r' = 0 \Leftrightarrow \cos\dfrac{\theta}{2} = 0 \Leftrightarrow \dfrac{\theta}{2} = \dfrac{\pi}{2} + k\pi \Leftrightarrow \theta = \pi + 2k\pi,  \forall k \in \mathbb{Z} $\\

$ \rightarrow \left\{\pi, 3\pi\right\} $\\
