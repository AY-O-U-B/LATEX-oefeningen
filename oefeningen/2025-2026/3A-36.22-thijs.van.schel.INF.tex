$f(x) = \ln\left(\sqrt[3]{3x+\sqrt{9x^2+1}}\right)$\\

$f(x) = \ln\left(\left(3x+\sqrt{9x^2+1}\right)^{\tfrac{1}{3}}\right)$\\

$f(x) = \dfrac{1}{3} \ln\left(3x+\sqrt{9x^2+1}\right)$\\

$f'(x) = \dfrac{1}{3} \cdot \dfrac{1}{3x+\sqrt{9x^2+1}} \cdot \left(3x+\sqrt{9x^2+1}\right)'$\\

$\left(3x+\sqrt{9x^2+1}\right)' = 3 + \dfrac{1}{2\sqrt{9x^2+1}} \cdot 18x$\\

$\left(3x+\sqrt{9x^2+1}\right)' = 3 + \dfrac{9x}{\sqrt{9x^2+1}}$\\

$\left(3x+\sqrt{9x^2+1}\right)' = \dfrac{3\sqrt{9x^2+1} + 9x}{\sqrt{9x^2+1}}$\\

$\left(3x+\sqrt{9x^2+1}\right)' = \dfrac{3(3x + \sqrt{9x^2+1})}{\sqrt{9x^2+1}}$\\

$f'(x) = \dfrac{1}{3} \cdot \dfrac{1}{3x+\sqrt{9x^2+1}} \cdot \dfrac{3(3x + \sqrt{9x^2+1})}{\sqrt{9x^2+1}}$\\

$f'(x) = \dfrac{1}{\sqrt{9x^2+1}}$\\
