$f(x)=\dfrac{x-1}{(x+1)^2}$\\

\begin{enumerate}

\item Domein\\

$Dom f=\mathbb{R}\setminus\{-1\}$\\


\item Asymptoten\\

Verticale asymptoot:\\

$x+1=0$\\

$x=-1$\\

$\lim\limits_{x \rightarrow -1^-}\dfrac{x-1}{(x+1)^2}=-\infty$\\

$\lim\limits_{x \rightarrow -1^+}\dfrac{x-1}{(x+1)^2}=-\infty$\\

$\Rightarrow$ VA: $x=-1$\\


Horizontale asymptoot:\\

$\lim\limits_{x \rightarrow +\infty}\dfrac{x-1}{(x+1)^2}=0$\\

$\lim\limits_{x \rightarrow -\infty}\dfrac{x-1}{(x+1)^2}=0$\\

$\Rightarrow$ HA: $y=0$\\


\item Tekenonderzoek van $f(x)$\\

Teller nul:\\

$x-1=0$\\

$x=1$\\

$\Rightarrow f^{(-1)}(0)=\{1\}$\\


\begin{tabular}{c|ccccc}

$x$ & $-\infty$ & $-1$ & $1$ &  & $+\infty$ \\ \hline

$f(x)$ & $-$ & $|$ & $0$ & $+$ & $+$ \\

\end{tabular}\\


\item Tekenonderzoek van $f'(x)$\\

$f(x)=\dfrac{x-1}{(x+1)^2}$\\

$f'(x)=\dfrac{(x+1)^2\cdot 1-(x-1)\cdot 2(x+1)}{(x+1)^4}$\\

$f'(x)=\dfrac{(x+1)[(x+1)-2(x-1)]}{(x+1)^4}$\\

$f'(x)=\dfrac{(x+1)(3-x)}{(x+1)^4}$\\

$f'(x)=\dfrac{3-x}{(x+1)^3}$\\


Teller nul:\\

$3-x=0$\\

$x=3$\\

$\Rightarrow f'^{(-1)}(0)=\{3\}$\\


\begin{tabular}{c|ccccc}

$x$ & $-\infty$ & $-1$ & $3$ &  & $+\infty$ \\ \hline

$f'(x)$ & $-$ & $|$ & $+$ & $0$ & $-$ \\

\end{tabular}\\


Maximum:\\

$f(3)=\dfrac{3-1}{(3+1)^2}$\\

$f(3)=\dfrac{2}{16}$\\

$f(3)=\dfrac{1}{8}$\\

$\Rightarrow M\left(3,\dfrac{1}{8}\right)$\\


\item Tekenonderzoek van $f''(x)$\\

$f'(x)=\dfrac{3-x}{(x+1)^3}$\\

$f''(x)=\dfrac{-1\cdot (x+1)^3-(3-x)\cdot 3(x+1)^2}{(x+1)^6}$\\

$f''(x)=\dfrac{(x+1)^2[-(x+1)-3(3-x)]}{(x+1)^6}$\\

$f''(x)=\dfrac{2x-10}{(x+1)^4}$\\


Teller nul:\\

$2x-10=0$\\

$x=5$\\

$\Rightarrow f''^{(-1)}(0)=\{5\}$\\


\begin{tabular}{c|ccccc}

$x$ & $-\infty$ & $-1$ & $5$ &  & $+\infty$ \\ \hline

$f''(x)$ & $-$ & $|$ & $-$ & $0$ & $+$ \\

\end{tabular}\\


Buigpunt:\\

$f(5)=\dfrac{5-1}{(5+1)^2}$\\

$f(5)=\dfrac{4}{36}$\\

$f(5)=\dfrac{1}{9}$\\

$\Rightarrow B\left(5,\dfrac{1}{9}\right)$\\
