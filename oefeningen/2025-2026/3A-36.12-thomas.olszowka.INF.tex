\begin{eqnarray*}
f(x)    & = & x \ln x - x \\
f'(x)   & = & 1 \ln x + x \dfrac{1}{x} - 1  \\
        & = & \ln x + \dfrac{x}{x} - 1 \\
        & = & \ln x + 1 - 1 \\
        & = & \ln x  
\end{eqnarray*}
