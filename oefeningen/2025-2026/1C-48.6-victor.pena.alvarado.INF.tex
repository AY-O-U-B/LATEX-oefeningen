$z^6 = -64 = -64 + 0i \Rightarrow
 \left\{
\begin{array}{l}
x=-64\\
y = 0
\end{array}
\right .$\\

$r = \sqrt{(-64)^2 + 0^2} = 64$ \\

$y=0, x<0 \Rightarrow \theta = \pi$

\begin{eqnarray*}
w_0 & = & \sqrt[6]{64}\cis\left(\dfrac{\pi + (0)2\pi}{6}\right) \\
& = & 2\cis\left(\dfrac{\pi}{6}\right) \\
& = & 2\left(\cos\dfrac{\pi}{6} + i\sin\dfrac{\pi}{6}\right) \\
& = & 2\left(\dfrac{\sqrt{3}}{2} + \dfrac{1}{2}i\right) \\
& = & \sqrt{3} + i
\end{eqnarray*}
\begin{eqnarray*}
w_1 & = & \sqrt[6]{64}\cis\left(\dfrac{\pi + (1)2\pi}{6}\right) \\

& = & 0 + 2i \\
& = & 2i
\end{eqnarray*}\begin{eqnarray*}

\end{eqnarray*}\begin{eqnarray*}

\end{eqnarray*}\begin{eqnarray*}
w_4 & = & \sqrt[6]{64}\cis\left(\dfrac{\pi + (4)2\pi}{6}\right) \\
