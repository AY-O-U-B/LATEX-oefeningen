$\displaystyle I=\int \dfrac{x^2+7x+1}{(x+2)^2(2x+1)} dx$\\

$\displaystyle \dfrac{x^2+7x+1}{(x+2)^2(2x+1)}=\dfrac{A}{(x+2)^2}+\dfrac{B}{x+2}+\dfrac{C}{2x+1}$\\

$\displaystyle A=\dfrac{x^2+7x+1}{2x+1}|_{x=-2}=\dfrac{4-14+1}{-4+1}=\dfrac{-9}{-3}=3$\\

$\displaystyle \dfrac{x^2+7x+1}{(x+2)^2(2x+1)}=\dfrac{3}{(x+2)^2}+\dfrac{B}{x+2}+\dfrac{C}{2x+1}$\\

$\displaystyle \dfrac{x^2+7x+1}{(x+2)^2(2x+1)}-\dfrac{3}{(x+2)^2}=\dfrac{x^2+7x+1-3(2x+1)}{(x+2)^2(2x+1)}$\\

$\displaystyle \dfrac{x-1}{(x+2)(2x+1)}=\dfrac{B}{x+2}+\dfrac{C}{2x+1}$\\

$\displaystyle B=\dfrac{x-1}{2x+1}|_{x=-2}=\dfrac{-3}{-3}=1$\\

$\displaystyle C=\dfrac{x-1}{x+2}|_{x=-\dfrac{1}{2}}=\dfrac{-\dfrac{3}{2}}{\dfrac{3}{2}}=-1$\\

$\displaystyle \dfrac{x^2+7x+1}{(x+2)^2(2x+1)}=\dfrac{3}{(x+2)^2}+\dfrac{1}{x+2}-\dfrac{1}{2x+1}$\\

$\displaystyle I=3\int\dfrac{dx}{(x+2)^2}+\int\dfrac{dx}{x+2}-\int\dfrac{dx}{2x+1}$\\
