\newcommand{\Bgtan}{\operatorname{Bgtan}}

\medskip

\normalsize

$2\Bgtan\dfrac{1}{2} - \Bgtan\dfrac{1}{7}$\\

$= \Bgtan\left(\dfrac{2\cdot\dfrac{1}{2}}{1-\left(\dfrac{1}{2}\right)^2}\right) - \Bgtan\dfrac{1}{7}$\\

$= \Bgtan \left(\dfrac{4}{3}\right) - \Bgtan \left(\dfrac{1}{7}\right)$\\

$=\Bgtan\left(\dfrac{\dfrac{4}{3}-\dfrac{1}{7}}{1+\dfrac{4}{3}\cdot \dfrac{1}{7}}\right)$\\

$=\Bgtan\left(\dfrac{\dfrac{25}{21}}{\dfrac{25}{21}}\right)$\\

$=\Bgtan(1)$\\

$= \dfrac{\pi}{4}$
