\left\{

\end{array}
\right.$\\

$\Rightarrow
\left\{
\begin{array}{l}
-12a-2b=-22\\

c=30-16a-4b
\end{array}
\right.$\\

$\Rightarrow
\left\{
\begin{array}{l}

\end{array}
\right.$\\

$\Rightarrow
\left\{
\begin{array}{l}

\end{array}
\right.$\\

$\Rightarrow
\left\{
\begin{array}{l}

\end{array}
\right.$\\

\left\{
\begin{array}{l}
