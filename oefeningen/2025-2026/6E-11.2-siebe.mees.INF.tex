$f'(x)=5x^4-40x^3+120x^2-160x+80$\\

$f''(x)=20x^3-120x^2+240x-160$\\

$f'''(x)=60x^2-240x+240$\\

$f^{IV}(x)=120x-240$\\

$f^{V}(x)=120$\\

$f'(2)=0$\\

$f''(2)=0$\\

$f'''(2)=0$\\

$f^{iv}(2)=0$\\

$f^{v}(2)=120\neq 0$\\
