$f(x) = \ln\left(\sqrt{\dfrac{1+\cos(ax)}{1-\cos(ax)}}\right)$\\

$f(x) = \ln\left(\left(\dfrac{1+\cos(ax)}{1-\cos(ax)}\right)^{\tfrac{1}{2}}\right)$\\

$f(x) = \dfrac{1}{2} \ln\left(\dfrac{1+\cos(ax)}{1-\cos(ax)}\right)$\\

$f(x) = \dfrac{1}{2} \left( \ln(1+\cos(ax)) - \ln(1-\cos(ax)) \right)$\\

$f'(x) = \dfrac{1}{2} \left( \dfrac{1}{1+\cos(ax)} \cdot (-a\sin(ax)) - \dfrac{1}{1-\cos(ax)} \cdot (a\sin(ax)) \right)$\\

$f'(x) = \dfrac{-a\sin(ax)}{2} \left( \dfrac{1}{1+\cos(ax)} + \dfrac{1}{1-\cos(ax)} \right)$\\

$f'(x) = \dfrac{-a\sin(ax)}{2} \left( \dfrac{(1-\cos(ax)) + (1+\cos(ax))}{(1+\cos(ax))(1-\cos(ax))} \right)$\\

$f'(x) = \dfrac{-a\sin(ax)}{2} \left( \dfrac{2}{1-\cos^2(ax)} \right)$\\

$f'(x) = \dfrac{-a\sin(ax)}{2} \cdot \dfrac{2}{\sin^2(ax)}$\\

$f'(x) = \dfrac{-a}{\sin(ax)}$\\
