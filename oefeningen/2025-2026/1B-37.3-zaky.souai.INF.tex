$ 3 - 4i $\\

$\left\{
    \begin{array}{lll} 
        x^2 - y^2 &=& 3 \\
        2xy &=& -4
    \end{array}
\right .$\\

$\Rightarrow \left\{
    \begin{array}{lll} 
        x^2 - y^2 &=& 3 \\
        xy &=& -2
    \end{array}
\right .$\\

$\Leftrightarrow \left\{
    \begin{array}{lll} 
        x^2 + (-y^2) &=& 3 \\
        x^2 \cdot (-y^2) &=& -4 \\
        xy < 0
    \end{array}
\right .$\\

$\Rightarrow \lambda^2 - 3\lambda - 4 = 0 $ \\

$\Rightarrow \lambda_1 = \dfrac{-b + \sqrt{9 + 16}}{2} = 4 $\\

$\Rightarrow \lambda_2 = \dfrac{-b - \sqrt{9 + 16}}{2} = -1  $\\

$\Rightarrow x = \sqrt{4} $ en $ y = \sqrt{-1} $\\
