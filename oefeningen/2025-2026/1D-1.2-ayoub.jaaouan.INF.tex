$ a(z+i) + b(2z+1+i) = iz +1 $\\

$ \Rightarrow az + ai + 2bz + b + bi = iz + 1 $\\

$\Rightarrow\left\{
	\begin{array}{lll} 
	    az+2bz &=& iz \\
		ai+b+bi &=& 1
	\end{array}
\right .$\\

$\Rightarrow\left\{
	\begin{array}{lll} 
	    a+2b &=& i \\
		ai+b(1+i) &=& 1
	\end{array}
\right .$\\

$\Rightarrow\left\{
	\begin{array}{lll} 
	    a &=& i-2b \\
		(i-2b)i+b(1+i) &=& 1
	\end{array}
\right .$\\

$\Rightarrow\left\{
	\begin{array}{lll} 
	    a &=& i-2b \\
		-1-2bi+b+bi &=& 1
	\end{array}
\right .$\\

$\Rightarrow\left\{
	\begin{array}{lll} 
	    a &=& i-2b \\
		1+b-bi &=& 1
	\end{array}
\right .$\\

$\Rightarrow\left\{
	\begin{array}{lll} 
	    a &=& i-2b \\
		b(1-i) &=& 2
	\end{array}
\right .$\\

$\Rightarrow\left\{
	\begin{array}{lll} 
	    a &=& i-2b \\
		b &=& \dfrac{2}{1-i}
	\end{array}
\right .$\\

$\Rightarrow\left\{
	\begin{array}{lll} 
	    a &=& i-2(1+i) \\
		b &=& 1+i
	\end{array}
\right .$\\

$\Rightarrow\left\{
	\begin{array}{lll} 
	    a &=& -2-i \\
		b &=& 1+i
	\end{array}
\right .$\\
