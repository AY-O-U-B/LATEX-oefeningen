$p(z)=2z^{3}+(3-2i)z^{2}+(1-3i)z-i$\\

\begin{enumerate}
\item Nulpunt zoeken.\\

$p(i)=2i^{3}+(3-2i)i^{2}+(1-3i)i-i $\\

$=2(-i)+(3-2i)(-1)+(1-3i)i-i $\\

$=-2i+(-3+2i)+(i+3)-i $\\

$=0 $\\

Dus $z=i$ is een nulpunt.\\

\item Factoriseren met $(z-i) $\\

$p(z)=(z-i)\left(2z^{2}+3z+1\right) $\\

\item Verdere factorisatie van $2z^{2}+3z+1 $ (of gebruik discriminant) \\

$2z^{2}+3z+1=(2z+1)(z+1) $\\
\end{enumerate}

$ \Rightarrow p(z)=(z-i)(2z+1)(z+1) $\\

De nulpunten zijn $z=i$, $z=-1$, $z=-\dfrac{1}{2} $\\
