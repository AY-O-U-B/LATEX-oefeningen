$f(x) = \dfrac{\sqrt{2x^3-2x+1}}{\sqrt[3]{x+\dfrac{1}{x}}}$\\

$f'(x) = \dfrac{\left(\sqrt{2x^3-2x+1}\right)' \cdot \sqrt[3]{x+\dfrac{1}{x}} - \sqrt{2x^3-2x+1} \cdot \left(\sqrt[3]{x+\dfrac{1}{x}}\right)'}{\left(\sqrt[3]{x+\dfrac{1}{x}}\right)^2}$\\

$\left(\sqrt{2x^3-2x+1}\right)' = \dfrac{1}{2\sqrt{2x^3-2x+1}} \cdot (6x^2-2) = \dfrac{3x^2-1}{\sqrt{2x^3-2x+1}}$\\

$\left(\sqrt[3]{x+\dfrac{1}{x}}\right)' = \dfrac{1}{3\left(x+\dfrac{1}{x}\right)^{\tfrac{2}{3}}} \cdot \left(1-\dfrac{1}{x^2}\right) = \dfrac{x^2-1}{3x^2\sqrt[3]{\left(x+\dfrac{1}{x}\right)^2}}$\\

$f'(x) = \dfrac{\dfrac{3x^2-1}{\sqrt{2x^3-2x+1}} \cdot \sqrt[3]{x+\dfrac{1}{x}} - \sqrt{2x^3-2x+1} \cdot \dfrac{x^2-1}{3x^2\sqrt[3]{\left(x+\dfrac{1}{x}\right)^2}}}{\sqrt[3]{\left(x+\dfrac{1}{x}\right)^2}}$\\

$f'(x) = \dfrac{(3x^2-1) \cdot 3x^2 \cdot \left(x+\dfrac{1}{x}\right) - (2x^3-2x+1)(x^2-1)}{3x^2\sqrt{2x^3-2x+1}\sqrt[3]{\left(x+\dfrac{1}{x}\right)^2} \cdot \sqrt[3]{\left(x+\dfrac{1}{x}\right)^2}}$\\

$f'(x) = \dfrac{(3x^2-1) \cdot 3x^2 \cdot \left(\dfrac{x^2+1}{x}\right) - (2x^5-2x^3-2x^3+2x+x^2-1)}{3x^2\sqrt{2x^3-2x+1}\left(x+\dfrac{1}{x}\right)^{\tfrac{4}{3}}}$\\

$f'(x) = \dfrac{3x(3x^2-1)(x^2+1) - (2x^5-4x^3+x^2+2x-1)}{3x^2\sqrt{2x^3-2x+1}\sqrt[3]{\left(x+\dfrac{1}{x}\right)^4}}$\\

$f'(x) = \dfrac{3x(3x^4+2x^2-1) - 2x^5+4x^3-x^2-2x+1}{3x^2\sqrt{2x^3-2x+1}\sqrt[3]{\left(x+\dfrac{1}{x}\right)^4}}$\\

$f'(x) = \dfrac{9x^5+6x^3-3x - 2x^5+4x^3-x^2-2x+1}{3x^2\sqrt{2x^3-2x+1}\sqrt[3]{\left(x+\dfrac{1}{x}\right)^4}}$\\

$f'(x) = \dfrac{7x^5+10x^3-x^2-5x+1}{3x^2\sqrt{2x^3-2x+1}\sqrt[3]{\left(x+\dfrac{1}{x}\right)^4}}$\\
