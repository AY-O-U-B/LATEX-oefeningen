\textbf{Definities:}
\begin{align*}
    M(A) &= \{x \in \mathbb{R} : \forall a \in A : a \leq x\} \\
    m(A) &= \{x \in \mathbb{R} : \forall a \in A : x \leq a\} \\
    \inf A &= a \Leftrightarrow 
        \big( \forall b \in A : a \leq b \big) 
        \text{ en } 
        \big( \forall \varepsilon > 0, \exists b \in A : b \leq a + \varepsilon \big) \\
    \min A &= a \Leftrightarrow a = \inf A \text{ en } a \in A \\
    \sup A &= b \Leftrightarrow 
        \big( \forall a \in A : a \leq b \big) 
        \text{ en } 
        \big( \forall \varepsilon > 0, \exists a \in A : b - \varepsilon \leq a \big) \\
    \max A &= b \Leftrightarrow b = \sup A \text{ en } b \in A
\end{align*}

\noindent (* indien $A \neq \emptyset$)

\bigskip

\textbf{1A - 4.6:} Bereken $M(A)$; $m(A)$; $\inf A$; $\min A$; $\sup A$; $\max A$ voor:

\[
A = \{ x \in \mathbb{R}^+ : \ln(x) = \dfrac{1}{n}
\text{ voor zekere } n \in \mathbb{N} \}
\]

\textbf{Observaties:}
\begin{itemize}
    \item $\ln(x) = \frac{1}{n}$ impliceert $x = e^{1/n}$.
    \item Hoe groter $n$, hoe kleiner $x$.
    \item $\displaystyle \lim_{n \to \infty} e^{1/n} = e^0 = 1$.
    \item Voor $n = 1$ is $x = e$.
\end{itemize}

\textbf{Oplossing:}\\

$A = \{e, e^{1/2}, e^{1/3}, e^{1/4}, \dots\}$\\

\begin{eqnarray*}
M(A) & = & \left[ e, +\infty \right[ \\
m(A) & = & \left] -\infty, 1 \right] \\
\inf A & = & 1 \\
\min A & \text{ bestaat niet} &  \\
\sup A & = &  e \\
\max A & = & e
\end{eqnarray*}
