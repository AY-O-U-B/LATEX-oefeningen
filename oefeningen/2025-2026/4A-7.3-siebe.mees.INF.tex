$ \displaystyle\int \dfrac{-x^2\sqrt{x} - \sqrt[4]{x^3}}{x^{\tfrac{1}{3}}}dx $\\

$ = \displaystyle\int \dfrac{-x^2\sqrt{x}}{x^{\tfrac{1}{3}}}dx - \displaystyle\int \dfrac{\sqrt[4]{x^3}}{x^{\tfrac{1}{3}}}dx $\\

$ = \displaystyle\int \dfrac{-x^2 x^{\tfrac{1}{2}}}{x^{\tfrac{1}{3}}}dx - \displaystyle\int \dfrac{x^{\tfrac{3}{4}}}{x^{\tfrac{1}{3}}}dx $\\

$ = \displaystyle\int -x^{\tfrac{5}{2}-\tfrac{1}{3}}dx - \displaystyle\int x^{\tfrac{3}{4}-\tfrac{1}{3}}dx $\\

$ = \displaystyle\int -x^{\tfrac{13}{6}}dx - \displaystyle\int x^{\tfrac{5}{12}}dx $\\

$ = -\dfrac{6}{19}x^{\tfrac{19}{6}} - \dfrac{12}{17}x^{\tfrac{17}{12}} + C $\\
