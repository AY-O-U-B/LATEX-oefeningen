\begin{eqnarray*}
(2\operatorname{cis}10^\circ)^4 (\operatorname{cis}16^\circ)^5
&=& 2^4 \operatorname{cis}(4\cdot 10^\circ) \operatorname{cis}(5\cdot 16^\circ) \\
&=& 16 \operatorname{cis}40^\circ \operatorname{cis}80^\circ \\
&=& 16 \operatorname{cis}(40^\circ + 80^\circ) \\
&=& 16 \operatorname{cis}120^\circ
\end{eqnarray*}

\begin{eqnarray*}
16 \operatorname{cis}120^\circ
&=& 16\left(\cos120^\circ + i\sin120^\circ\right) \\
&=& 16\left(-\dfrac{1}{2} + i\dfrac{\sqrt{3}}{2}\right) \\
&=& -8 + 8\sqrt{3}i
\end{eqnarray*}
