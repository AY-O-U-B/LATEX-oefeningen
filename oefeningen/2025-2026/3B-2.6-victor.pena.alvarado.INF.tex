	$f(x)=\dfrac{1}{x^2+2x}$\\
	
	$f'(x) = -\dfrac{2x+2}{(x^2 +2x)^2}$\\
	
	$f''(x) =  -1\cdot\left[\dfrac{2(x^2+2x)^2-(2x+2)2(x^2+2x)(2x+2)}{(x^2 +2x)^4}\right]$ \\
	
	$=-1\cdot\left[\dfrac{2(x^2+2x)((x^2+2x)-(2x+2)^2)}{(x^2+2x)^4}\right]=-1\cdot\left[\dfrac{2(x^2+2x-4x^2-8x - 4)^2)}{(x^2 +2x)^3}\right]$ \\ 
	
	$=\dfrac{2(3x^2 + 6x +4)}{(x^2 +2x)^3}$ \\
	
	$f'''(x) = 2\left[\dfrac{(x^2+2x)^3(6x+6)-(3x^2+6x+4)3(x^2+2x)^2(2x+2)}{(x^2+2x)^6}\right]$ \\
	
	$=2\left[\dfrac{(6x+6)(x^2+2x)^2((x^2+2x)-(3x^2+6x+4))}{(x^2+2x)^6}\right]$ \\
	
	$=2\left[\dfrac{(6x+6)((x^2+2x)-(3x^2+6x+4))}{(x^2+2x)^4}\right]$ \\
	
	$=2\left[\dfrac{(6x+6)(-2x^2-4x-4)}{(x^2+2x)^4}\right]=-24\left[\dfrac{(x+1)(x^2+2x+2)}{(x^2+2x)^4}\right]=\dfrac{-24(x^3+3x^2+4x+2)}{(x^2+2x)^4}$
