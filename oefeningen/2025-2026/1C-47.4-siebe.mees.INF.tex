$ z^5 = -i $\\

\begin{enumerate}

\item Poolvorm \\

$ \left \{
\begin{array}{l}
    r = \sqrt{(0^2 + (-1)^2} = 1 \\
    \theta = \tan^{-1} \left( \dfrac{0}{-1} \right) + \pi = -\dfrac{\pi}{2}
\end{array}
\right .$\\ 

$ \Rightarrow -i = 1 \cis \left(- \dfrac{\pi}{2} \right) $\\

\item Vijfdemachtswortels \\

$ w = \sqrt[n]{r} \cis \left( \dfrac{\theta + k 2 \pi}{n} \right), k \in {0,1,2,..., n-1} $\\

$ = \sqrt[5]{1} \cis \left( \dfrac{- \dfrac{\pi}{2} + k 2 \pi}{5} \right) $\\

$ = \cis \left( -\dfrac{\pi}{10} + \dfrac{2k\pi}{5} \right) $\\

$ \Rightarrow \left\{  \cis \left( - \dfrac{\pi}{10} \right),  \cis \left( \dfrac{3\pi}{10} \right), \cis \left( \dfrac{7\pi}{10} \right), \cis \left( \dfrac{11\pi}{10} \right), -i \right\} $\\ 

Note: $\cis\left(\dfrac{3\pi}{2}\right)=-i $\\

\end{enumerate}
