\noindent \textbf{Schets de functie:} $r = 3(1 - \sin(\theta))$\\

$\left .
\begin{array}{l}
    \operatorname{dom} r    = \mathbb{R}\\
    P                       = 2\pi
\end{array}
\right \}
\text{Bep. Dom.} = [0, 2\pi]$\\

\noindent \textbf{Nulpunten:}\\

$r = 0 \Leftrightarrow \sin(\theta) = 0 \Leftrightarrow \theta = \dfrac{\pi}{2}+ k2\pi$ met $k \in \mathbb{N} \longrightarrow \theta \in \left\{\dfrac{\pi}{2}\right\}$\\

$r' = 0 \Leftrightarrow -3\cos(\theta) = 0 \Leftrightarrow \cos(\theta) = 0 \Leftrightarrow \theta = \dfrac{\pi}{2} + k\pi$ met $k \in \mathbb{N}$\\

$\longrightarrow \theta \in \left\{ \dfrac{\pi}{4}, \dfrac{3\pi}{4}\right\}$\\
