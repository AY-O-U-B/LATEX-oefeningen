$f(x,y) = x^3 + x^2y^3 - 3y^4$\\

$f(1,-1) = (1)^3 + (1)^2(-1)^3 - 3(-1)^4 = 1 - 1 - 3 = -3$\\

$
\left\{ 
\begin{array}{l} 
\dfrac{\partial f}{\partial x}(x,y) = 3x^2 + 2xy^3 \\[3ex] 
\dfrac{\partial f}{\partial y}(x,y) = 3x^2y^2 - 12y^3 
\end{array} 
\right.
\Rightarrow
\left\{ 
\begin{array}{l} 
\dfrac{\partial f}{\partial x}(1,-1) = 3(1)^2 + 2(1)(-1)^3 = 1 \\[3ex] 
\dfrac{\partial f}{\partial y}(1,-1) = 3(1)^2(-1)^2 - 12(-1)^3 = 15 
\end{array} 
\right.
$ \\

$
\left\{ 
\begin{array}{l} 
\dfrac{\partial^2 f}{\partial x^2}(x,y) = 6x + 2y^3 \\[3ex] 
\dfrac{\partial^2 f}{\partial x \partial y}(x,y) = 6xy^2 \\[3ex]
\dfrac{\partial^2 f}{\partial y^2}(x,y) = 6x^2y - 36y^2
\end{array} 
\right.
\Rightarrow
\left\{ 
\begin{array}{l} 
\dfrac{\partial^2 f}{\partial x^2}(1,-1) = 6(1) + 2(-1)^3 = 4 \\[3ex] 
\dfrac{\partial^2 f}{\partial x \partial y}(1,-1) = 6(1)(-1)^2 = 6 \\[3ex]
\dfrac{\partial^2 f}{\partial y^2}(1,-1) = 6(1)^2(-1) - 36(-1)^2 = -42
\end{array} 
\right.
$ \\

$T_2(x,y) = f(a,b) + \dfrac{\partial f}{\partial x}(a,b)(x-a) + \dfrac{\partial f}{\partial y}(a,b)(y-b)$\\

$+ \dfrac{1}{2!} \left[ \dfrac{\partial^2 f}{\partial x^2}(a,b)(x-a)^2 + 2\dfrac{\partial^2 f}{\partial x \partial y}(a,b)(x-a)(y-b) + \dfrac{\partial^2 f}{\partial y^2}(a,b)(y-b)^2 \right]$\\

$\Rightarrow T_2(x,y) = -3 + 1(x-1) + 15(y+1) + \dfrac{1}{2} \left[ 4(x-1)^2 + 2(6)(x-1)(y+1) - 42(y+1)^2 \right]$\\

$= -3 + (x-1) + 15(y+1) + 2(x-1)^2 + 6(x-1)(y+1) - 21(y+1)^2$\\

$= -14 + 3x - 33y + 2x^2 - 21y^2 + 6xy$\\
