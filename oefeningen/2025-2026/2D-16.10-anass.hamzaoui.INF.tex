\begin{eqnarray*}
    && \lim\limits_{x\rightarrow \tfrac{\pi}{2}} \dfrac{1-\cos 4x}{(2x-\pi)\cdot\sin 2x} \\
    && = \lim\limits_{x\rightarrow \tfrac{\pi}{2}} \dfrac{1-1+2\cdot\sin^22x}{(2x-\pi)\cdot\sin 2x} \\
    && = \lim\limits_{x\rightarrow \tfrac{\pi}{2}} \dfrac{2\cdot\sin^22x}{(2x-\pi)\cdot\sin 2x} \\
    && = \lim\limits_{x\rightarrow \tfrac{\pi}{2}} \dfrac{2\cdot\sin2x}{(2x-\pi)} \\
    && = \lim\limits_{x\rightarrow \tfrac{\pi}{2}} \dfrac{-2\cdot\sin(2x-\pi)}{(2x-\pi)} \\
    && = -2
\end{eqnarray*}
