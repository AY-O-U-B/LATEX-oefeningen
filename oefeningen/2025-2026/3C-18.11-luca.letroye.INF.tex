$ \lim\limits_{x \rightarrow 0} \dfrac{x^2\ln(1+x)}{\ln (1-x)} = \dfrac{0}{0}$\\

$ \overset{H}{=}  \lim\limits_{x \rightarrow 0} \dfrac{2x\ln(1+x)+x^2\cdot \dfrac{1}{1+x}}{\cos x - 1}$\\

$ =  \lim\limits_{x \rightarrow 0} \dfrac{2x\ln(1+x)+\dfrac{x^2}{1+x}}{\cos x - 1}$\\

$  \overset{H}{=}  \lim\limits_{x \rightarrow 0} \dfrac{2\ln(1+x)+\dfrac{2x+2x^2+2x+2x^2-x^2}{(1+x)^2}}{-\sin x}$\\

$  = \lim\limits_{x \rightarrow 0} \dfrac{2\ln(1+x)+\dfrac{4x+3x^2}{(1+x)^2}}{-\sin x}$\\

$  \overset{H}{=} \lim\limits_{x \rightarrow 0} \dfrac{\dfrac{2}{1+x}+\dfrac{(4+6x)(1+2x) - 2(4x+3x^2)}{(1+x)^3}}{-\cos x}$\\

$  = \lim\limits_{x \rightarrow 0} \dfrac{\dfrac{2}{1+x}+\dfrac{2x+4}{(1+x)^3}}{-\cos x}$\\

$  = \lim\limits_{x \rightarrow 0} \dfrac{\dfrac{2x^2+6x+6}{(1+x)^3}}{-\cos x} = \dfrac{6}{-1} = -6$\\
