$f(x) = \sin(2x)$\\

$f'(x) = 2\cos(2x)$\\

$f'(x) = 2\sin\left(2x+\dfrac{\pi}{2}\right)$\\

$f''(x) = 2^2\cos\left(2x+\dfrac{\pi}{2}\right)$\\

$f''(x) = 2^2\sin\left(2x+\pi\right)$\\

$f'''(x) = 2^3\cos\left(2x+\pi\right)$\\

$f'''(x) = 2^3\sin\left(2x+\dfrac{3\pi}{2}\right)$\\

$\vdots$\\

$f^{(n)}(x) = 2^n\sin\left(2x+\dfrac{n\pi}{2}\right)$\\
