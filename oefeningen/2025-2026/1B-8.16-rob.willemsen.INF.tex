\begin{eqnarray*}
  (5-4i)(3+i) & = & 5 \cdot 3 + 5 \cdot i - 4i \cdot 3 - 4i \cdot i \\
              & = & 15 + 5i - 12i - 4i^2 \\
              & = & 15 + 5i - 12i - 4 \cdot (-1) \\
              & = & 15 + 5i - 12i + 4 \\
              & = & 19 - 7i
\end{eqnarray*}

\textbf{Antwoord:} $19 - 7i$
