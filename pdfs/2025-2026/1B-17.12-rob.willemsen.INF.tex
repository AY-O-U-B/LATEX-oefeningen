\begin{eqnarray*}
  (2-i)^3 & = & 2^3 - 3 \cdot 2^2 \cdot i + 3 \cdot 2 \cdot i^2 - i^3 \\
          & = & 8 - 12i + 3 \cdot 2 \cdot (-1) - (-i) \\
          & = & 8 - 12i - 6 + i \\
          & = & 2 - 11i
\end{eqnarray*}

\begin{flushleft}
\textbf{Antwoord:} $2 - 11i$
