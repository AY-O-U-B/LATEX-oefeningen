$f(x)=(1+e^{2x})^{-1}$ \\

$f'(x)=-1(1+e^{2x})^{-2}(e^{2x})'$ \\

$=-1(1+e^{2x})^{-2}2e^{2x}$ \\

$=-2e^{2x}(1+e^{2x})^{-2}$ \\

$f''(x)=-2\left[(e^{2x})'(1+e^{2x})^{-2}+e^{2x}(-2)(1+e^{2x})^{-3}(e^{2x})'\right]$ \\

$=-2\left[2e^{2x}(1+e^{2x})^{-2}-4e^{4x}(1+e^{2x})^{-3}\right]$ \\

$=-4e^{2x}(1+e^{2x})^{-2}+8e^{4x}(1+e^{2x})^{-3}$ \\

$=4e^{2x}(e^{2x}-1)(1+e^{2x})^{-3}$ \\

\fbox{$f''(x)=4e^{2x}(e^{2x}-1)(1+e^{2x})^{-3}$} \\
