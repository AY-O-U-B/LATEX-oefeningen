\noindent \textbf{Schets de functie:} $r = 1 + \cos(\theta)$\\

$\left .
\begin{array}{l}
    \operatorname{dom} r    = \mathbb{R}\\
    P                       = \pi
\end{array}
\right \}
\text{Bep. Dom.} = [0, \pi]$\\

\noindent \textbf{Nulpunten:}\\

$r = 0 \Leftrightarrow \sin(2\theta) = 0 \Leftrightarrow 2\theta = k\pi$ met $k \in \mathbb{N} \longrightarrow \theta \in \left\{0, \dfrac{\pi}{2}, \pi\right\}$\\

$r' = 0 \Leftrightarrow 6\sin(2\theta) = 0 \Leftrightarrow \cos(2\theta) = 0 \Leftrightarrow 2\theta = \dfrac{\pi}{2} + k\pi$ met $k \in \mathbb{N}$\\

$\longrightarrow \theta \in \left\{ \dfrac{\pi}{4}, \dfrac{3\pi}{4}\right\}$\\
