$\theta_k = \dfrac{\pi+2k\pi}{4}, \quad k \in \{0, 1, 2, 3\}$\\



$k = 0$ \\

$\theta_0 = \dfrac{\pi}{4}$ \\

$z_0 = \sqrt{2} \left(\cos \dfrac{\pi}{4} + i \sin \dfrac{\pi}{4}\right)$ \\

$z_0 = \sqrt{2} \left(\dfrac{\sqrt{2}}{2} + i \dfrac{\sqrt{2}}{2}\right)$ \\

$z_0 = 1 + i$\\



$k = 1$ \\

$\theta_1 = \dfrac{3\pi}{4}$ \\

$z_1 = \sqrt{2} \left(\cos \dfrac{3\pi}{4} + i \sin \dfrac{3\pi}{4}\right)$ \\

$z_1 = \sqrt{2} \left(-\dfrac{\sqrt{2}}{2} + i \dfrac{\sqrt{2}}{2}\right)$ \\

$z_1 = -1 + i$\\



$k = 2$ \\

$\theta_2 = \dfrac{5\pi}{4}$ \\

$z_2 = \sqrt{2} \left(\cos \dfrac{5\pi}{4} + i \sin \dfrac{5\pi}{4}\right)$ \\

$z_2 = \sqrt{2} \left(-\dfrac{\sqrt{2}}{2} - i \dfrac{\sqrt{2}}{2}\right)$ \\

$z_2 = -1 - i$\\



$k = 3$ \\

$\theta_3 = \dfrac{7\pi}{4}$ \\

$z_3 = \sqrt{2} \left(\cos \dfrac{7\pi}{4} + i \sin \dfrac{7\pi}{4}\right)$ \\

$z_3 = \sqrt{2} \left(\dfrac{\sqrt{2}}{2} - i \dfrac{\sqrt{2}}{2}\right)$ \\

$z_3 = 1 - i$\\
