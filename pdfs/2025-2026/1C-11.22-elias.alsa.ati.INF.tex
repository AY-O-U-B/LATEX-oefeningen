$$z = \dfrac{1}{2} - \dfrac{\sqrt{3}}{2}i $$ \\

\textbf{ Berekening van de modulus ($r = |z|$): }
\begin{align*}
    r   &= \sqrt{a^2 + b^2}                                                             \\
    r   &= \sqrt{ \left(\frac{1}{2}\right)^2 + \left(\dfrac{\sqrt{3}}{2}\right)^2 }    \\
    r   &= \sqrt{\dfrac{1}{4} + \dfrac{3}{4}}                                           \\
    r   &= \sqrt{1} = 1
\end{align*}

\textbf{ Berekening van het argument ($\alpha$): }
\begin{align*}
    \tan \alpha &= \dfrac{  \dfrac{-\sqrt{3}}{2}}{\dfrac{1}{2}} \\
    \tan \alpha &= \dfrac{- \sqrt{3}}{2} \cdot 2                \\
    \tan \alpha &= \dfrac{-2 \sqrt{3}}{2}                       \\
    \alpha      &= \operatorname{Bgtan} \left(-\sqrt{3}\right)               \\
    \alpha      &= - \dfrac{\pi}{3}
\end{align*}

\textbf{ De geometrische vorm ($z = r \cdot (\cos \alpha + i \sin \alpha)$): }
$$ z = \cos \left(-\dfrac{\pi}{3}\right) + i \sin \left(-\dfrac{\pi}{3}\right)$$
