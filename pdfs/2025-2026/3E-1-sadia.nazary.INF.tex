$y^2=2x$ \\

$x=\dfrac{y^2}{2}$ \\

$d^2=(x-1)^2+(y-4)^2$ \\

$d^2=\left(\dfrac{y^2}{2}-1\right)^2+(y-4)^2$ \\

$f(y)=\left(\dfrac{y^2}{2}-1\right)^2+(y-4)^2$ \\

$f'(y)=2\left(\dfrac{y^2}{2}-1\right)\left(\dfrac{y^2}{2}-1\right)'+2(y-4)(y-4)'$ \\

$\left(\dfrac{y^2}{2}-1\right)'=y$ \\

$(y-4)'=1$ \\

$f'(y)=2\left(\dfrac{y^2}{2}-1\right)y+2(y-4)$ \\

$f'(y)=y^3-2y+2y-8$ \\

$f'(y)=y^3-8$ \\

$f'(y)=0$ \\

$y^3-8=0$ \\

$y=2$ \\

$x=\dfrac{y^2}{2}$ \\

$x=\dfrac{4}{2}$ \\

$x=2$ \\
