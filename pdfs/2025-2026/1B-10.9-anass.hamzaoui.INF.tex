\begin{eqnarray*}
    && (i - 1)(i - 2)(i + 1)(i + 2) \\
    && = (i^2 - 2i - i + 2)(i + 1)(i + 2) \\
    && = (i^2 - 3i + 2)(i + 1)(i + 2) \\
    && = (-1 - 3i + 2)(i + 1)(i + 2) \\
    && = (1 - 3i)(i + 1)(i + 2) \\
    && = (i + 1 - 3i^2 - 3i)(i + 2) \\
    && = (-2i + 1 - 3(-1))(i + 2) \\
    && = (-2i + 4)(i + 2) \\
    && = -2i^2 - 4i + 4i + 8 \\
    && = -2(-1) + 8 \\
    && = 2 + 8 \\
    && = 10 \\
\end{eqnarray*}
