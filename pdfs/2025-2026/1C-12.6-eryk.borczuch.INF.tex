\normalsize

$\dfrac{-1}{2+2i}$\\

teller: \\

modulus = $1$\\

argument = $1$\\


noemer:\\

modulus = $\sqrt{2^2+2^2}$\\

$= \sqrt{8} = 2 \sqrt{2}$\\

argument = $\operatorname{Bgtan} \left(\dfrac{2}{2}\right)$\\

$= \dfrac{\pi}{4}$\\

geheel:\\

modulus = $\dfrac{1}{2\sqrt{2}} = \dfrac{\sqrt{2}}{4}$\\

argument = $\pi - \dfrac{\pi}{4} = \dfrac{3 \pi}{4}$\\

goniometrische vorm = $ \dfrac{\sqrt{2}}{4} \operatorname{cis} \left(\dfrac{3\pi}{4}\right)$
