$
\left\{
    \begin{array}{lll} 
        x^2 - y^2 &=& -1\\
        2xy &=& -2\sqrt{2}
    \end{array}
\right.
\Leftrightarrow
\left\{
    \begin{array}{lll} 
        x^2 - y^2 &=& -1\\
        x^2y^2 &=& 2\\
    \end{array}
\right.
\Leftrightarrow
\left\{
    \begin{array}{lll} 
        x^2 - y^2 &=& -1\\
        x^2(-y)^2 &=& -2\\
    \end{array}
\right.
$\\

$\Rightarrow \lambda^2 + \lambda - 2 = 0$\\

$D = 1^2 - 4 \cdot 1 \cdot (-2) = 9$\\
