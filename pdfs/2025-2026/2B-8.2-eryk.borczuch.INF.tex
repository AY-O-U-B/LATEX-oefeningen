\newcommand{\Bgtan}{\operatorname{Bgtan}}

\medskip

\normalsize

$\Bgtan\dfrac{1}{2} = \Bgtan\dfrac{1}{3} + \Bgtan\dfrac{1}{7}$\\

$= \Bgtan\left(\dfrac{\dfrac{1}{3}+\dfrac{1}{7}}{1-\dfrac{1}{3}\cdot\dfrac{1}{7}}\right)$\\

$=\Bgtan \left(\dfrac{\dfrac{10}{21}}{\dfrac{20}{21}}\right)$\\

$= \Bgtan\left(\dfrac{1}{2}\right)$
